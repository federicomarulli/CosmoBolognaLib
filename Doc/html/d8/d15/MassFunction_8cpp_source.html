45 double cbl::cosmology::Cosmology::mass_function (const double Mass, const double redshift, const string model_MF, const string method_SS, const string output_root, const double Delta, const string interpType, const int norm, const double k_min, const double k_max, const double prec, const string input_file, const bool is_parameter_file, const bool default_delta, const double delta_t)
46 {

49

51

147 double cbl::cosmology::Cosmology::m_MF_generator (const double Mass, const double Sigma, const double Dln_Sigma, const double redshift, const string model_MF, const double Delta, const bool default_delta, const double delta_t)
148 {
149  const double deltacz = (default_delta) ? deltac(redshift) : fabs(delta_t*DD(redshift)/DD(0.));
150  const double sigmaz = Sigma*DD(redshift)/DD(0.);
151 
152  const double RHO = rho_m(0., true);
153 
154  if (model_MF=="PS") // Press & Schechter
155  return sqrt(2./par::pi)*RHO/(Mass*Mass)*deltacz/sigmaz*fabs(Dln_Sigma)*exp(-(deltacz*deltacz)*0.5/(sigmaz*sigmaz));
156 
157  else if (model_MF=="ST") { // Sheth & Tormen
158  double aa = 0.707;
159  double qq = 0.3;
160  double AA = 0.3222;
161  return AA*sqrt(2.*aa/par::pi)*RHO/(Mass*Mass)*deltacz/sigmaz*(1+pow(sigmaz/(sqrt(aa)*deltacz),2.*qq))*fabs(Dln_Sigma)*exp(-(aa*deltacz*deltacz)*0.5/(sigmaz*sigmaz));
162  }
163 
164  else if (model_MF=="Jenkins") // Jenkins et al. (2001)
165  return 0.315*exp(-pow(fabs(log(1./sigmaz)+0.61),3.8))*RHO/(Mass*Mass)*fabs(Dln_Sigma);
166 
167  else if (model_MF=="Warren") { // Warren et al. (2006)
168  double AA = 0.7234;
169  double aa = 1.625;
170  double bb = 0.2538;
171  double cc = 1.1982;
172  return AA*(pow(sigmaz,-aa)+bb)*exp(-cc/(sigmaz*sigmaz))*RHO/(Mass*Mass)*fabs(Dln_Sigma);
173  }
174 
175  else if (model_MF=="Reed") { // Reed et al. (2007)
176  double cc = 1.08;
177  double aa = 0.764/cc;
178  double pp = 0.3;
179  double AA = 0.3222;
180  double Fatt = AA*sqrt(2.*aa/par::pi);
181  double n_eff = -6.*Dln_Sigma-3.;
182  double G1 = exp(-pow(log(1./sigmaz)-0.4,2)/0.72);
183  double G2 = exp(-pow(log(1./sigmaz)-0.75,2)/0.08);
184  return Fatt*RHO/(Mass*Mass)*deltacz/sigmaz*(1+pow(sigmaz*sigmaz/(aa*deltacz*deltacz),pp)+0.6*G1+0.4*G2)*fabs(Dln_Sigma)*exp(-(cc*aa*deltacz*deltacz)*0.5/(sigmaz*sigmaz)-(0.03/pow(n_eff+3.,2)*pow(deltacz/sigmaz,0.6)));
185  }
186 
187  else if (model_MF=="Pan") { // Pan (2007)
188  double alpha = 0.435; // check!!!
189  return 4.*alpha/sqrt(2.*par::pi)*RHO/(Mass*Mass)*deltacz/pow(sigmaz,2.*alpha)*fabs(Dln_Sigma)*exp(-(deltacz*deltacz)*0.5/pow(sigmaz,4.*alpha));
190  }
191 
192  else if (model_MF=="ShenH") { // halo MF, Shen et al. (2006) // check!!!
193  double alpha = -0.55;
194  double beta = -0.56;
195  double ni = pow(deltacz/sigmaz,2);
196  double ni_fni = sqrt(ni/(2.*par::pi))*exp(-ni*pow(1.-beta*pow(ni,alpha),2)*0.5)*(1.-beta/pow(ni,-alpha)*(1+alpha+(alpha*(alpha+1.))*0.5));
197  return ni_fni*RHO/(Mass*Mass)*fabs(Dln_Sigma)*2;
198  }
199 
200  else if (model_MF=="ShenF") { // filament MF, Shen et al. (2006) // check!!!
201  double alpha = -0.28;
202  double beta = -0.012;
203  double ni = pow(deltacz/sigmaz,2);
204  double ni_fni = sqrt(ni/(2.*par::pi))*exp(-ni*pow(1.-beta*pow(ni,alpha),2)*0.5)*(1.-beta/pow(ni,-alpha)*(1+alpha+(alpha*(alpha+1.))*0.5));
205  return ni_fni*RHO/(Mass*Mass)*fabs(Dln_Sigma)*2;
206  }
207 
208  else if (model_MF=="ShenS") { // sheet MF, Shen et al. (2006) // check!!!
209  double alpha = -0.61;
210  double beta = 0.45;
211  double ni = pow(deltacz/sigmaz,2);
212  double ni_fni = sqrt(ni/(2.*par::pi))*exp(-ni*pow(1.-beta*pow(ni,alpha),2)*0.5)*(1.-beta/pow(ni,-alpha)*(1+alpha+(alpha*(alpha+1.))*0.5));
213  return fabs(ni_fni*RHO/(Mass*Mass)*fabs(Dln_Sigma)*2); // check!!!
214  }
215 
216  else if (model_MF=="Tinker") { // Tinker et al. (2008)
217 
218  //if (redshift>2) WarningMsg("Attention: the Tinker mass function has been tested for z<~2!");
219 
220  double A0, a0, b0, c0;
221 
222  if (Delta==200) {A0 = 0.186; a0 = 1.47; b0 = 2.57; c0 = 1.19;}
223  else if (Delta==300) {A0 = 0.200; a0 = 1.52; b0 = 2.25; c0 = 1.27;}
224  else if (Delta==400) {A0 = 0.212; a0 = 1.56; b0 = 2.05; c0 = 1.34;}
225  else if (Delta==600) {A0 = 0.218; a0 = 1.61; b0 = 1.87; c0 = 1.45;}
226  else if (Delta==800) {A0 = 0.248; a0 = 1.87; b0 = 1.59; c0 = 1.58;}
227  else if (Delta==1200) {A0 = 0.255; a0 = 2.13; b0 = 1.51; c0 = 1.80;}
228  else if (Delta==1600) {A0 = 0.260; a0 = 2.30; b0 = 1.46; c0 = 1.97;}
229  else if (Delta==2400) {A0 = 0.260; a0 = 2.53; b0 = 1.44; c0 = 2.24;}
230  else if (Delta==3200) {A0 = 0.260; a0 = 2.66; b0 = 1.41; c0 = 2.44;}
231  else {
232  A0 = (Delta<1600) ? 0.1*log10(Delta)-0.05 : 0.26;
233  a0 = 1.43+pow(max(log10(Delta)-2.3,0.),1.5); // check!!!
234  b0 = 1.+pow(log10(Delta)-1.6,-1.5);
235  c0 = 1.2+pow(max(log10(Delta)-2.35,0.),1.6); // check!!!
236  }
237 
238  double alpha = pow(10.,-pow(max(0.75/log10(Delta/75.),0.),1.2)); // check!!!
239  double AA = A0*pow(1.+redshift,-0.14);
240  double aa = a0*pow(1.+redshift,-0.06);
241  double bb = b0*pow(1.+redshift,-alpha);
242  double cc = c0;
243 
244  return AA*(pow(sigmaz/bb,-aa)+1.)*exp(-cc/(sigmaz*sigmaz))*RHO/(Mass*Mass)*fabs(Dln_Sigma);
245  }
246 
247  else if (model_MF=="Tinker08_Interp") { // Tinker et al. (2008)
248 
249  //if (redshift>2) WarningMsg("Attention: the Tinker mass function has been tested for z<~2!");
250 
251  vector<double> _Delta = {200, 300, 400, 600, 800, 1200, 1600, 2400, 3200};
252  vector<double> _A0 = {1.858659e-01, 1.995973e-01, 2.115659e-01, 2.184113e-01, 2.480968e-01, 2.546053e-01, 2.600000e-01, 2.600000e-01, 2.600000e-01};
253  vector<double> _a0 = {1.466904, 1.521782, 1.559186, 1.614585, 1.869936, 2.128056, 2.301275, 2.529241, 2.661983};
254  vector<double> _b0 = {2.571104, 2.254217, 2.048674, 1.869559, 1.588649, 1.507134, 1.464374, 1.436827, 1.405210};
255  vector<double> _c0 = {1.193958, 1.270316, 1.335191, 1.446266, 1.581345, 1.795050, 1.965613, 2.237466, 2.439729};
256 
257  cbl::glob::FuncGrid interp_A0(_Delta, _A0, "Spline");
258  cbl::glob::FuncGrid interp_a0(_Delta, _a0, "Spline");
259  cbl::glob::FuncGrid interp_b0(_Delta, _b0, "Spline");
260  cbl::glob::FuncGrid interp_c0(_Delta, _c0, "Spline");
261 
262  double alpha = pow(10.,-pow(max(0.75/log10(Delta/75.),0.),1.2)); // check!!!
263  double AA = interp_A0(Delta)*pow(1.+redshift,-0.14);
264  double aa = interp_a0(Delta)*pow(1.+redshift,-0.06);
265  double bb = interp_b0(Delta)*pow(1.+redshift,-alpha);
266  double cc = interp_c0(Delta);
267 
268  return AA*(pow(sigmaz/bb,-aa)+1.)*exp(-cc/(sigmaz*sigmaz))*RHO/(Mass*Mass)*fabs(Dln_Sigma);
269  }
270 
271 
272  else if (model_MF=="Crocce") { // Crocce et al. (2010)
273  double AA = 0.58*pow(1.+redshift,-0.13);
274  double aa = 1.37*pow(1.+redshift,-0.15);
275  double bb = 0.3*pow(1.+redshift,-0.084);
276  double cc = 1.036*pow(1.+redshift,-0.024);
277  return AA*(pow(sigmaz,-aa)+bb)*exp(-cc*pow(sigmaz,-2))*RHO/(Mass*Mass)*fabs(Dln_Sigma);
278  }
279 
280  else if (model_MF=="Angulo_FOF") // FoF MF, Angulo et al. (2012)
281  return 0.201*(pow(2.08/sigmaz,1.7)+1)*exp(-1.172/(sigmaz*sigmaz))*RHO/(Mass*Mass)*fabs(Dln_Sigma);
282 
283  else if (model_MF=="Angulo_Sub") // SUBFIND MF, Angulo et al. (2012)
284  return 0.265*(pow(1.675/sigmaz,1.9)+1)*exp(-1.4/(sigmaz*sigmaz))*RHO/(Mass*Mass)*fabs(Dln_Sigma);
285 
286  else if (model_MF=="Bhattacharya") { // Bhattacharya et al. (2011) // check!!!
287  double aa = 0.788/pow(1+redshift,0.01);
288  double qq = 1.795;
289  double AA = 0.333/pow(1+redshift,0.11);
290  double pp = 0.807;
291  double nu = deltacz/sigmaz;
292  //if (redshift>=2) aa = 0.799/pow(1+redshift,0.024);
293  return RHO/(Mass*Mass)*fabs(Dln_Sigma)*AA*sqrt(2/par::pi)*exp(-nu*nu*0.5*aa)*(1+pow(1/(nu*nu*aa),pp))*pow(nu*sqrt(aa),qq);
294  }
295 
296  else if (model_MF=="Courtin") { // Courtin et al. (2010)
297  double aa = 0.695;
298  double qq = 0.1;
299  double AA = 0.348;
300  return AA*sqrt(2.*aa/par::pi)*RHO/(Mass*Mass)*deltacz/sigmaz*(1+pow(sigmaz/(sqrt(aa)*deltacz),2.*qq))*fabs(Dln_Sigma)*exp(-(aa*deltacz*deltacz)*0.5/(sigmaz*sigmaz));
301  }
302 
303  else if (model_MF=="Manera") { // Manera et al. (2010)
304 
305  double aa, qq;
306  if (fabs(redshift-0.)<1.e-30) {
307  aa = 0.709;
308  qq = 0.248;
309  }
310  else if (fabs(redshift-0.5)<1.e-30) {
311  aa = 0.724;
312  qq = 0.241;
313  }
314  else
315  ErrorCBL("Error in cbl::cosmology::Cosmology::m_MF_generator of MassFunction.cpp! The Manera mass function has been tested only for z=0 and 0.5!");
316 
317  return 0.3222*sqrt(2.*aa/par::pi)*RHO/(Mass*Mass)*deltacz/sigmaz*(1+pow(sigmaz/(sqrt(aa)*deltacz),2.*qq))*fabs(Dln_Sigma)*exp(-(aa*deltacz*deltacz)*0.5/(sigmaz*sigmaz));
318  }
319 
320  else if (model_MF=="Watson_FOF") { // FoF MF, Watson et al. (2012)
321  double a = 0.282;
322  double b = 1.406;
323  double c = 2.163;
324  double d = 1.210;
325  return a*(pow((b/sigmaz),c)+1)*exp(-d/(sigmaz*sigmaz))*RHO/(Mass*Mass)*fabs(Dln_Sigma);
326  }
327 
328  else if (model_MF=="Watson_SOH") { // Spherical Overdensity halo MF, Watson et al. (2012)
329  double Omega = OmegaM(redshift);
330  double C = exp(0.023*(Delta/178-1));
331  double d = -0.456*Omega-0.139;
332  double p = 0.072;
333  double q = 2.13;
334  double Gamma = C*pow((Delta/178),d)*exp(p*(1-Delta/178)/pow(sigmaz,q));
335  double A = 0.194;
336  double alpha = 2.267;
337  double beta = 1.805;
338  double gamma = 1.287;
339  if (redshift>0 && redshift<6) {
340  A = Omega*(1.097*pow((1+redshift),-3.216)+0.074);
341  alpha = Omega*(3.136*pow((1+redshift),-3.056)+2.349);
342  beta = Omega*(5.907*pow((1+redshift),-3.599)+2.344);
343  gamma = 1.318;
344  }
345  if (redshift>=6) {
346  A = 0.563;
347  alpha = 0.874;
348  beta = 3.810;
349  gamma = 1.453;
350  }
351  return RHO/(Mass*Mass)*fabs(Dln_Sigma)*Gamma*A*(pow((beta/sigmaz),alpha)+1.)*exp(-gamma/(sigmaz*sigmaz));

354  else if (model_MF=="Peacock") { // Peacock et al. (2007)
355  double aa = 1.529;
356  double bb = 0.704;
357  double cc = 0.412;
358  double nu = deltacz/sigmaz;

362  else return ErrorCBL("Error in cbl::cosmology::Cosmology::m_MF_generator of MassFunction.cpp: model_MF not allowed!");
363 }

369 double cbl::cosmology::Cosmology::n_haloes (const double Mass_min, const double Mass_max, const double z_min, const double z_max, const bool angle_rad, const string model_MF, const string method_SS, const string output_root, const double Delta, const string interpType, const double k_max, const string input_file, const bool is_parameter_file)
370 {
371 
372  // ---------- read the grid file ----------
373 
374  const string file_grid = create_grid_sigmaM(method_SS, 0., output_root,interpType, k_max, input_file, is_parameter_file);
375 
376  ifstream fin(file_grid.c_str()); checkIO(fin, file_grid);
377 
378  double Mass, Sigma, Dln_Sigma;
379  vector<double> mass, sigma, dlnsigma;
380  while (fin >>Mass>>Sigma>>Dln_Sigma) {
381  if (Mass_min<Mass && Mass<Mass_max) {
382  mass.push_back(Mass);
383  sigma.push_back(Sigma);
384  dlnsigma.push_back(Dln_Sigma);
385  }
386  }
387 
388 
389  // ---------- compute the number of haloes ----------
390 
391  int step_z = 100;
392  double delta_z = (z_max-z_min)/step_z;
393  double redshift = z_min;
394  double N_haloes = 0.;
395 
396  for (int Red=0; Red<step_z; Red++) {
397 
398  double Int = 0.;
399  for (unsigned int k=0; k<mass.size()-1; k++)
400  Int += mass_function(mass[k], sigma[k], dlnsigma[k], redshift, model_MF, output_root, Delta)*dV_dZdOmega(redshift, angle_rad)*(mass[k+1]-mass[k]);
401 
402  N_haloes += Int*delta_z;
403 
404  redshift += delta_z;
405  }
406 
407  return N_haloes;
408 }

414 double cbl::cosmology::Cosmology::n_haloes (const double Mass_min, const double Mass_max, const double Volume, const double redshift, const string model_MF, const string method_SS, const int nbin_mass, const string output_root, const double Delta, const string interpType, const int norm, const double k_min, const double k_max, const double prec, const string input_file, const bool is_parameter_file, const bool default_delta, const double delta_t)
415 {

417 
418  mass_function_par.redshift = redshift;
419  mass_function_par.model_MF = model_MF ;
420  mass_function_par.method_SS = method_SS;
421  mass_function_par.output_root = output_root;
422  mass_function_par.Delta = Delta;
423  mass_function_par.interpType = interpType;
424  mass_function_par.norm = norm;
425  mass_function_par.k_min = k_min;
426  mass_function_par.k_max = k_max;
427  mass_function_par.prec = prec;

431  mass_function_par.delta_t = delta_t;
432 
433  auto pp = make_shared<glob::STR_MF>(mass_function_par);

435  if (nbin_mass==0) {
436  function<double(double)> func = bind(&Cosmology::m_mass_function, this, std::placeholders::_1, pp);

452 
453  else

455

457 }
