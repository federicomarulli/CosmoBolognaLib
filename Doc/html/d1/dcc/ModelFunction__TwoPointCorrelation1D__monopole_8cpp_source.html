56

59 
60  vector<double> new_rad;
61

668 std::vector<double> cbl::modelling::twopt::xi0_linear_cosmology_clusters_selection_function (const std::vector<double> rad, const std::shared_ptr<void> inputs, std::vector<double> &parameter)
669 {
670  // structure contaning the required input data
671  shared_ptr<STR_data_model> pp = static_pointer_cast<STR_data_model>(inputs);
672 
673  // set the test cosmology
674  cosmology::Cosmology cosmo = *pp->test_cosmology;
675 
676 
677  // ----- input likelihood parameters -----
678 
679  // set the cosmological parameters used to compute the dark matter
680  // two-point correlation function in real space
681  for (size_t i=0; i<pp->Cpar.size(); ++i)
682  cosmo.set_parameter(pp->Cpar[i], parameter[i]);
683 
684  // set the alpha parameter of the cluster mass scaling relation
685  const double alpha = parameter[pp->Cpar.size()];
686 
687 
688  // set the function to estimate the linear dark matter power spectrum at z=0, by interpolating it from a grid
689 
690  const vector<double> Pk_grid = cosmo.Pk(pp->kk, pp->method_Pk, false, 0., pp->output_dir, pp->store_output_CAMB, pp->output_root, -1, pp->k_min, pp->k_max, pp->prec, pp->file_par);
691  glob::FuncGrid interp_Pk(pp->kk, Pk_grid, "Spline");
692 
693 
694  // compute sigma and dlnsigma, by interpolating them from a grid
695 
696  // set the cluster masses and rho_m
697  const vector<double> mass = pp->mass;
698  const double rho = cosmo.rho_m(0.);
699 
700  vector<double> sigma_grid, dnsigma_grid;
701 
702  for (size_t i=0; i<mass.size(); i++) {
703 
704  const double RR = Radius(mass[i], rho);
705 
706  // compute sigma
707  auto func_sigma = [&] (double kk)
708  {
709  return pow(TopHat_WF(kk*RR)*kk, 2)*interp_Pk(kk);
710  };
711  sigma_grid.emplace_back(sqrt(1./(2.*pow(par::pi, 2))*wrapper::gsl::GSL_integrate_qag(func_sigma, pp->k_min, pp->k_max, pp->prec)));
712 
713  // compute dlnsigma
714  const double dRdM = pow(3./(4.*par::pi*rho), 1./3.)*pow(mass[i], -2./3.)/3.;
715  auto func_dnsigma = [&] (const double kk)
716  {
717  double filter = 2*TopHat_WF(kk*RR)*TopHat_WF_D1(kk*RR)*kk*dRdM;
718  return filter*pow(kk, 2)*interp_Pk(kk);
719  };
720  dnsigma_grid.emplace_back(1./(2.*pow(par::pi, 2))*wrapper::gsl::GSL_integrate_qag(func_dnsigma, pp->k_min, pp->k_max, pp->prec));
721 
722  }
723 
724  glob::FuncGrid interp_sigma(mass, sigma_grid, "Spline");
725  glob::FuncGrid interp_DnSigma(mass, dnsigma_grid, "Spline");
726 
727 
728  // compute the bias
729  const double bias = cosmo.bias_eff_selection_function(interp_sigma, interp_DnSigma, *pp->interp_SelectionFunction_cut, pp->Mass_min, pp->Mass_max, {pp->redshift}, pp->model_bias, pp->model_MF, "EisensteinHu", alpha, pp->store_output_CAMB, pp->output_root, pp->Delta, -1., "Spline", pp->norm, pp->k_min, pp->k_max, pp->prec)[0]; // check!!!
730  parameter[pp->Cpar.size()+1] = bias;
731 
732  // set the AP factor
733  const double AP_factor = cosmo.D_V(pp->redshift)/pp->DVfid;
734 
735  // rescale with the AP factor
736  vector<double> new_rad = rad;
737  for (size_t i=0; i<rad.size(); i++)
738  new_rad[i] *= AP_factor;
739 
740  // compute the real-space monopole of the two-point correlation function at z=0, by Fourier transforming the P(k)
741  vector<double> xi = wrapper::fftlog::transform_FFTlog(new_rad, 1, pp->kk, Pk_grid, 0);
742 
743  // compute the redshift-space monopole at z=pp->redshift
744  const double fact = pow(bias, 2)*xi_ratio(cosmo.linear_growth_rate(pp->redshift, 1), bias)*pow(cosmo.DD(pp->redshift)/cosmo.DD(0), 2);
745  for (size_t i=0; i<xi.size(); i++)
746  xi[i] *= fact;
747 
748  return xi;
749 }
