75 double cbl::cosmology::Cosmology::m_bias_halo_generator (const double Sigma, const double redshift, const string author, const double Delta) const
76 {
77  const double deltacz = deltac(redshift);
78  const double sigmaz = Sigma*DD(redshift)/DD(0.);
79 
80  double bias = -1000.;
81 
82  if (author=="ST99") {
83  double aa = 0.707;
84  double pp = 0.3;
85  double ni = pow(deltacz/sigmaz, 2);
86  bias = 1.+(aa*ni-1.)/deltacz+(2.*pp/deltacz)/(1.+pow(aa*ni,pp));
87  }
88 
89  else if (author=="SMT01") {
90  double aa = 0.707;
91  double bb = 0.5;
92  double cc = 0.6;
93  double ni = deltacz/sigmaz;
94  bias = 1.+1./(sqrt(aa)*deltacz)*(sqrt(aa)*aa*pow(ni,2.)+sqrt(aa)*bb*pow(aa*pow(ni,2.),1.-cc)-pow(aa*pow(ni,2.),cc)/(pow(aa*pow(ni,2.),cc)+bb*(1.-cc)*(1.-cc*0.5)));
95  }
96 
97  else if (author=="SMT01_WL04") {
98  double aa = 0.707;
99  double bb = 0.5;
100  double cc = 0.6;
101  double ni = deltacz/sigmaz;
102  double niI = sqrt(aa)*ni;
103  bias = 1.+1./deltacz*(pow(niI,2.)+bb*pow(niI,2.*(1.-cc))-pow(niI,2.*cc)/sqrt(aa)/(pow(niI,2.*cc)+bb*(1.-cc)*(1.-cc*0.5)));
104  }
105 
106  else if (author=="Tinker") { // Tinker et al. (2010)
107  double yy = log10(Delta);
108  double AA = 1.+0.24*yy*exp(-pow(4./yy,4));
109  double aa = 0.44*yy-0.88;
110  double BB = 0.183;
111  double bb = 1.5;
112  double CC = 0.019+0.107*yy+0.19*exp(-pow(4./yy,4));
113  double ccc = 2.4;
114  double ni = 1.686/sigmaz;
115  bias = 1.-AA*pow(ni,aa)/(pow(ni,aa)+pow(1.686,aa))+BB*pow(ni,bb)+CC*pow(ni,ccc);
116  }
117 
118  else
119  ErrorCBL("Error in cbl::cosmology::Cosmology::m_bias_halo_generator() of Bias.cpp: author = " + author + "!");
120 
121  return bias;
122 }
