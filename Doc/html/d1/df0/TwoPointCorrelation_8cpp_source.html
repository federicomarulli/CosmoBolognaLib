185 void cbl::measure::twopt::TwoPointCorrelation::count_pairs (const shared_ptr<Catalogue> cat1, const ChainMesh_Catalogue &ChM, shared_ptr<Pair> pp, const bool cross, const bool tcount)
186 {
187  // timer
188  time_t start; time(&start);
189  int dp = cout.precision();
190  cout.setf(ios::fixed); cout.setf(ios::showpoint); cout.precision(2);
191 
192  // number of objects in the first catalogue
193  int nObj = cat1->nObjects();
194 
195  // factor used by the timer
196  float fact_count = 100./nObj;
197 
198  // pointer to the second catalogue (get from the chain mesh)
199  shared_ptr<Catalogue> cat2 = ChM.catalogue();
200 
201  // thread number
202  int tid = 0;
203 
204  // start the multithreading parallelization
205 #pragma omp parallel num_threads(omp_get_max_threads()) private(tid)
206  {
207 
208  tid = omp_get_thread_num();
209 
210  // if (tid == 0) coutCBL << "Number of threads = " << omp_get_num_threads() << endl;
211 
212 
213  // internal object used by each thread to handle pairs
214 
215  shared_ptr<Pair> pp_thread = (pp->pairDim()==Dim::_1D_) ? move(Pair::Create(pp->pairType(), pp->pairInfo(), pp->sMin(), pp->sMax(), pp->nbins(), pp->shift(), pp->angularUnits(), pp->angularWeight()))
216  : move(Pair::Create(pp->pairType(), pp->pairInfo(), pp->sMin_D1(), pp->sMax_D1(), pp->nbins_D1(), pp->shift_D1(), pp->sMin_D2(), pp->sMax_D2(), pp->nbins_D2(), pp->shift_D2(), pp->angularUnits(), pp->angularWeight()));
217 
218 
219  // parallelized loop
220 #pragma omp for schedule(static, 2)
221 
222  // loop on the objects of the first catalogue
223  for (int i=0; i<nObj; ++i) {
224 
225  // get the indexes of objects in the second catalogue that are close to the object i of the first catalogue
226  // (i.e. objects inside the close cells of the chain-mesh)
227  vector<long> close_objects = ChM.close_objects(cat1->coordinate(i), (cross) ? -1 : (long)i);
228 
229  // loop on the nearby objects
230  for (auto &&j : close_objects)
231  // estimate the distance between the two objects and update the pair count
232  pp_thread->put(cat1->catalogue_object(i), cat2->catalogue_object(j));
233 
234  // estimate the computational time and update the time count
235  time_t end_temp; time(&end_temp); double diff_temp = difftime(end_temp, start);
236  if (tcount && tid==0) { coutCBL << "\r" << float(i)*fact_count << "% completed (" << diff_temp << " seconds)\r"; cout.flush(); }
237  if (i==int(nObj*0.25)) coutCBL << ".............25% completed" << endl;
238  if (i==int(nObj*0.5)) coutCBL << ".............50% completed" << endl;
239  if (i==int(nObj*0.75)) coutCBL << ".............75% completed"<< endl;
240 
241  }
242 
243 #pragma omp critical
244  {
245  // sum all the object pairs computed by each thread
246  pp->Sum(pp_thread);
247  }
248 
249  }
250 
251 
252  // show the time spent by the method
253  time_t end; time (&end);
254  double diff = difftime(end, start);
255  if (tid==0) {
256  if (diff<60) coutCBL << " time spent to count the pairs: " << diff << " seconds" << endl ;
257  else if (diff<3600) coutCBL << " time spent to count the pairs: " << diff/60 << " minutes" << endl ;
258  else coutCBL << " time spent to count the pairs: " << diff/3600 << " hours" << endl ;
259  }
260  cout.unsetf(ios::fixed); cout.unsetf(ios::showpoint); cout.precision(dp);
261 
262 }

384 double cbl::measure::twopt::TwoPointCorrelation::PoissonError (const Estimator estimator, const double dd, const double rr, const double dr, const int nData, const int nRandom) const
385 {
386  if (estimator!=Estimator::_natural_ && estimator!=Estimator::_LandySzalay_)
387  ErrorCBL("The implementation of Poisson errors for the chosen estimator is not available yet!", glob::ExitCode::_workInProgress_);
388 
389  double fR = m_random_dilution_fraction;
390  if (estimator==Estimator::_natural_ && fR!=1) {
391  fR = 1.;
392  WarningMsg("Attention: --> fR = 1, since the random catalogue is not diluted when using the natural estimator!");
393  }
394 
395  const double norm1 = double(nRandom)*double(nRandom-1)/(double(nData)*double(nData-1));
396  const double norm2 = double(nRandom-1)/double(nData);
397 
398  const double T1 = pow(norm1*sqrt(dd)/rr, 2);
399  const double T2 = pow(norm2*sqrt(dr)/rr, 2);
400  const double T3 = pow((norm1*dd-norm2*dr)*pow(rr, -1.5), 2);
401 
402  if (min(T2, T3)>T1)
403  WarningMsg("Attention: enlarge the random sample, that dominates the Poisson errors!");
404 
405  return pow(fR, 2)*sqrt(T1+T2+T3);
406 }
