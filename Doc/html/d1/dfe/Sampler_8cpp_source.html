43 shared_ptr<random::DistributionRandomNumbers> cbl::statistics::Sampler::m_set_gz (const int seed, const double aa)
44 {
45  if (aa<=1)
46  ErrorCBL("the stretch move parameter must be >1!", "m_set_gz", "Sampler.cpp");
47 
48  double zmin = 1./aa;
49  double zmax = aa;
50  vector<double> zz = linear_bin_vector(1000,zmin,zmax), gzz;
51  for (auto &&zzz : zz)
52  gzz.push_back(1./sqrt(zzz));
53 
54  return make_shared<random::DistributionRandomNumbers>(random::DistributionRandomNumbers(zz, gzz, "Spline", seed));
55 }

118 void cbl::statistics::Sampler::sample_stretch_move (const int chain_size, const int nwalkers, const vector<vector<double>> start, const int seed, const double aa, const string outputFile)
119 {
120  set_chain(m_npar, m_npar_free, chain_size, nwalkers);
121 
122  function<void(int, int, const vector<double>, const double)> write_func;
123  ofstream fout;
124 
125  write_func = [&] (const int nc, const int nw, const vector<double> parameters, const double func)
126  {(void)nc; (void)nw; (void)parameters; (void)func;};
127 
128  if (outputFile!=par::defaultString) {
129  fout.open(outputFile.c_str());
130  write_func = [&] (const int nc, const int nw, const vector<double> parameters, const double func)
131  {
132  fout << nc << " " << nw << " ";
133  for (int p=0; p<m_npar; p++)
134  fout << parameters[p] << " ";
135  fout << func << endl;
136  };
137  }
138 
139  // set seed for random_numbers
140  random::UniformRandomNumbers_Int seeds(0, 23412432, seed);
141 
142  const int chain_seed = seeds();
143  const int MH_seed = seeds();
144  const int gz_seed = seeds();
145 
146  coutCBL << "Starting seed = " << seed << endl;
147  coutCBL << "Seed for random walker choice = " << chain_seed << endl;
148  coutCBL << "Seed for Metropolis-Hastings algorithm = " << MH_seed << endl;
149  coutCBL << "Seed for random extraction from g(z) = " << gz_seed << endl;
150 
151  random::UniformRandomNumbers_Int chains(0, m_nwalkers-1, chain_seed);
152  random::UniformRandomNumbers MH_random(0., 1., MH_seed);
153  auto gz = m_set_gz(gz_seed, aa);
154 
155  // initialize chains
156  m_initialize_chains(start);
157 
158  for (int n=1; n<m_chain_size; n++) {
159  for (int i=0; i<m_nwalkers; i++)
160  {
161  int kk = i;
162  while (kk==i)
163  kk = chains();
164 
165  vector<double> parameters_i;
166  vector<double> parameters_k;
167 
168  parameters_i = m_chains[n-1][i];
169  parameters_k = m_chains[n-1][kk];
170 
171  vector<double> parameters(m_npar, 0);
172  double proposed_function_chains = par::defaultDouble;
173 
174  double gen_z = gz->operator()();
175 
176  gen_z = gz->operator()();
177  for (int p=0; p<m_npar; p++)
178  parameters[p] = parameters_k[p] + gen_z*(parameters_i[p]-parameters_k[p]);
179  proposed_function_chains = m_function(parameters);
180 
181  parameters_i = parameters;
182 
183  double ratio = min(1.,pow(gen_z, m_npar_free-1)*exp(proposed_function_chains-m_function_chain[n-1][i]));
184 
185  if (MH_random() <ratio) {
186  m_function_chain[n][i] = proposed_function_chains;
187  m_chains[n][i] = parameters_i;
188  m_acceptance[i] += 1./m_chain_size;
189  }
190  else {
191  m_function_chain[n][i] = m_function_chain[n-1][i];
192  m_chains[n][i] = m_chains[n-1][i];
193  }
194 
195  write_func(n, i, m_chains[n][i], m_function_chain[n][i]);
196  }
197 
198  const int progress = int(double((n+1)*m_nwalkers)/(m_nwalkers*m_chain_size)*100);
199  coutCBL << progress << "% \r"; cout.flush();
200  }
201 
202  cout << endl;
203  coutCBL << "Done!" << endl;
204 }
