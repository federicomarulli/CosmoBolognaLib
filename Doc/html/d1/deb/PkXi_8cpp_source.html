56 double cbl::cosmology::Cosmology::sigma8_interpolated (const double redshift) const
57 {
58  double wm = m_Omega_matter*m_hh*m_hh;
59  double wb = m_Omega_baryon*m_hh*m_hh;
60  double sigma8 = 0.1058*pow(m_scalar_amp/2.196e-9, 0.5)*pow(wm/0.1426, 0.520)*pow(wb/0.02205, -0.294)*pow(m_hh/0.673, 0.683)*pow((m_massless_neutrinos+m_massive_neutrinos)/3.046, -0.24)*exp(0.3727*(m_n_spec-0.96))*pow(1-m_Omega_k, 0.175)*DD(redshift)/DD(9);
61 
62  return ((m_Omega_neutrinos>0) ? 0.995*sigma8 : sigma8);
63 }
