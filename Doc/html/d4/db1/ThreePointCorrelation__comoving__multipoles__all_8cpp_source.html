124  double jyy = cat->yy(close_objects[j]);
125  double jzz = cat->zz(close_objects[j]);
126  double jww = cat->weight(close_objects[j]);
127 
128  double xx = jxx-ixx;
129  double yy = jyy-iyy;
130  double zz = jzz-izz;
131 
132  double rr = sqrt(xx*xx+yy*yy+zz*zz);
133 
134  if (rr>=rmin && rr<=rmax && i!=close_objects[j]) {

175  }
176  }
177  }

183  int index = l+b1*norders+b1*norders*nbins;
184  NNN[index] -= NN[b1];
