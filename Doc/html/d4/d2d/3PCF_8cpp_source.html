564  l2.push_back(ll);

576  pk_r2p[i] = pk_sn[i]*jl(kk_sn[i]*r2_prime, l_prime);

582  vector<double> f_r_r2 = fftlog::transform_FFTlog(rr, 1, kk_sn, pk_r2, l, 0, kr0, 1);
583  vector<double> f_r_r1p = fftlog::transform_FFTlog(rr, 1, kk_sn, pk_r1p, l_prime, 0, kr0, 1);
584  vector<double> f_r_r2p = fftlog::transform_FFTlog(rr, 1, kk_sn, pk_r2p, l_prime, 0, kr0, 1);

586  glob::FuncGrid interp_f_r_r1(rr, f_r_r1, interpType);
587  glob::FuncGrid interp_f_r_r2(rr, f_r_r2, interpType);
588  glob::FuncGrid interp_f_r_r1p(rr, f_r_r1p, interpType);
589  glob::FuncGrid interp_f_r_r2p(rr, f_r_r2p, interpType);

598  }

601 
602  for (int ll=0; ll<n_l2; ll++) {
603 
604  vector<double> f_r_r1_r1p = fftlog::transform_FFTlog(rr, 1, kk_sn, pk_r1_r1p, l2[ll], 0, kr0, 1);
605  vector<double> f_r_r2_r2p = fftlog::transform_FFTlog(rr, 1, kk_sn, pk_r2_r2p, l2[ll], 0, kr0, 1);
606  vector<double> f_r_r2_r1p = fftlog::transform_FFTlog(rr, 1, kk_sn, pk_r2_r1p, l2[ll], 0, kr0, 1);
607  vector<double> f_r_r1_r2p = fftlog::transform_FFTlog(rr, 1, kk_sn, pk_r1_r2p, l2[ll], 0, kr0, 1);

609  interp_f_r_r2_r2p[ll] = make_shared<glob::FuncGrid> (glob::FuncGrid(rr, f_r_r2_r2p, interpType));
610  interp_f_r_r2_r1p[ll] = make_shared<glob::FuncGrid> (glob::FuncGrid(rr, f_r_r2_r1p, interpType));
611  interp_f_r_r1_r2p[ll] = make_shared<glob::FuncGrid> (glob::FuncGrid(rr, f_r_r1_r2p, interpType));
612  }
613 
614  // integrand
615 
616  auto integrand = [&] (const double rr)
617  {
618  double Xi_r = interp_Xi(rr);
619 
620  double f_r_r1 = interp_f_r_r1(rr);
621  double f_r_r2 = interp_f_r_r2 (rr);
622  double f_r_r1p = interp_f_r_r1p(rr);
623  double f_r_r2p = interp_f_r_r2p(rr);
624 
625  double sum = 0;
626 
627  for (int ll=0; ll<n_l2; ll++) {
628  double t1 = (2*l2[ll]+1)*pow(gsl_sf_coupling_3j(2*l, 2*l_prime, 2*l2[ll], 0, 0, 0),2);
629  if (t1>0) {
630  double f_r_r1_r1p = interp_f_r_r1_r1p[ll]->operator()(rr);
631  double f_r_r2_r2p = interp_f_r_r2_r2p[ll]->operator()(rr);
632  double f_r_r2_r1p = interp_f_r_r2_r1p[ll]->operator()(rr);
633  double f_r_r1_r2p = interp_f_r_r1_r2p[ll]->operator()(rr);
634 
635  double t2 = pow(-1, l2[ll])*Xi_r*(f_r_r1_r1p*f_r_r2_r2p+f_r_r2_r1p*f_r_r1_r2p);
636  double t3 = pow(-1, 0.5*(l+l_prime+l2[ll]))*(f_r_r1*f_r_r1p*f_r_r2_r2p+f_r_r1*f_r_r2p*f_r_r2_r1p+f_r_r2*f_r_r1p*f_r_r1_r2p+f_r_r2*f_r_r2p*f_r_r1_r1p);
637  sum += t1*(t2+t3);
638  }
639  }
640 
641  return rr*rr*sum;
642  };
643 
644  double fact = 4.*par::pi/Volume*(2*l+1)*(2*l_prime+1)*pow(-1, l+l_prime);
645 
646  return fact*gsl::GSL_integrate_qag(integrand, 0., 1000, prec);
