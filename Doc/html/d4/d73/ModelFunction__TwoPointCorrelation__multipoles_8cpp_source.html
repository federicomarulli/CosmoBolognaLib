187 std::vector<double> cbl::modelling::twopt::xiMultipoles_BAO (const std::vector<double> rad, const std::shared_ptr<void> inputs, std::vector<double> &parameter)
188 {
189  // structure contaning the required input data

191 
192  vector<double> Xi(rad.size());
193 
194  // input parameters
195 
196  // AP parameter that contains the distance information
197  double alpha_perpendicular = parameter[0];
198 
199  // AP parameter that contains the distance information
200  double alpha_parallel = parameter[1];
201 
202  for (size_t i=0; i<rad.size(); i++) {
203  const int mult = pp->dataset_order[i];
204  const int ell = 2*mult;
205  const double rr = rad[i];
206  const double fact = (2*ell+1);
207 
208  const double B = parameter[2+mult];
209  const double A0 = parameter[4+mult];
210  const double A1 = parameter[6+mult];
211  const double A2 = parameter[8+mult];
212 
213  auto integrand = [&] (const double mu_fid)
214  {
215  return twopt::Xi_polar(rr, mu_fid, alpha_perpendicular, alpha_parallel, pp->func_multipoles)*cbl::legendre_polynomial(mu_fid, ell);
216  };
217 
218  Xi[i] = B*fact*wrapper::gsl::GSL_integrate_qag(integrand, 0, 1)+A0+A1/rr+A2/(rr*rr);
219  }
220 
221  return Xi;
