192 std::vector<double> cbl::modelling::twopt::xiMultipoles_BAO (const std::vector<double> rad, const std::shared_ptr<void> inputs, std::vector<double> &parameter)
193 {
194  // structure contaning the required input data
195  shared_ptr<STR_data_model> pp = static_pointer_cast<STR_data_model>(inputs);
196 
197  vector<double> Xi(rad.size());
198 
199  // input parameters
200 
201  // AP parameter that contains the distance information
202  double alpha_perpendicular = parameter[0];
203 
204  // AP parameter that contains the distance information
205  double alpha_parallel = parameter[1];
206 
207  for (size_t i=0; i<rad.size(); i++) {
208  const int mult = pp->dataset_order[i];
209  const int ell = 2*mult;
210  const double rr = rad[i];
211  const double fact = (2*ell+1);
212 
213  const double B = parameter[2+mult];
214  const double A0 = parameter[4+mult];
215  const double A1 = parameter[6+mult];
216  const double A2 = parameter[8+mult];
217 
218  auto integrand = [&] (const double mu_fid)
219  {
220  return twopt::Xi_polar(rr, mu_fid, alpha_perpendicular, alpha_parallel, pp->func_multipoles)*cbl::legendre_polynomial(mu_fid, ell);
221  };
222 
223  Xi[i] = B*fact*wrapper::gsl::GSL_integrate_qag(integrand, 0, 1)+A0+A1/rr+A2/(rr*rr);
224  }
225 
226  return Xi;
227 }
