56 void cbl::data::Data1D::read (const string input_file, const int skip_nlines, const int column_x, const vector<int> column_data, const vector<int> column_errors)
57 {
58  if (column_data.size()!=column_errors.size()) ErrorCBL("the sizes of column_data and columt_errors must be equal!", "read", "Data1D.cpp");
59 
60  // default column of input data values
61  int column_data_default = column_x+1;
62 
63  // default column of input error values
64  int column_error_default = column_x+2;
65 
66  ifstream fin(input_file.c_str()); checkIO(fin, input_file);
67  string line;
68 
69  // loop on the data (and, possibly, error) vectors to be read; if
70  // more than one data vectors are read, the data (and, possibly,
71  // error) vectors will be added one after the other, in order to
72  // construct a single data object
73  const int cl_max = (column_data.size()<2) ? 1 : column_data.size();
74  for (int cl=0; cl<cl_max; ++cl) {
75 
76  // skip the first nlines (in case of header lines)
77  if (skip_nlines>0)
78  for (int i=0; i<skip_nlines; ++i)
79  getline(fin, line);
80 
81  // read the file lines
82  while (getline(fin, line)) {
83 
84  stringstream ss(line);
85  vector<double> num; double NUM;
86  while (ss>>NUM) num.emplace_back(NUM);
87 
88  if (num.size()<2) ErrorCBL("there are less than 2 columns in the input file!", "read", "Data1D.cpp");
89 
90  // if column_x is larger than 0, the column of x coordinates is
91  // the one specified in input
92  size_t ind_x = max(0, column_x-1);
93  checkDim(num, ind_x+1, "num", false);
94  m_x.emplace_back(num[ind_x]);
95 
96  // if the size of the column_data vector is larger than 1, the
97  // columns of data values are the ones specified in input
98  const size_t ind_data = (column_data.size()<1) ? column_data_default-1 : column_data[cl]-1;
99  checkDim(num, ind_data+1, "num", false);
100  m_data.emplace_back(num[ind_data]);
101 
102  // if the size of the column_errors vector is larger than 1, the
103  // columns of error values are the ones specified in input; if
104  // the error column is not present, the errors will be set to 1,
105  // by default
106  const size_t ind_error = (column_errors.size()<1) ? column_error_default-1 : column_errors[cl]-1;
107  if (num.size()<ind_error+1 && column_errors.size()>1)
108  WarningMsgCBL("the errors cannot be retrieved from the provided input file, and will be set to 1", "read", "Data1D.cpp");
109  m_error.emplace_back((num.size()<ind_error+1) ? 1 : num[ind_error]);
110 
111  }
112 
113  // go back at the beginning of the file
114  fin.clear(); fin.seekg(ios::beg);
115 
116  column_data_default += 2;
117  column_error_default += 2;
118  }
119 
120  fin.clear(); fin.close();
121 
122 
123  // set the data size and diagonal covariance
124 
125  m_ndata = m_data.size();
126 
127  m_covariance.resize(m_ndata, vector<double>(m_ndata, 0));
128  for (int i=0; i<m_ndata; i++)
129  m_covariance[i][i] = pow(m_error[i], 2);
130 
131 }

174 void cbl::data::Data1D::write_covariance (const string dir, const string file, const int precision) const
175 {
176  checkDim(m_covariance, m_ndata, m_ndata, "covariance", false);

178  string file_out = dir+file;

190  << " " << setiosflags(ios::fixed) << setprecision(precision) << setw(5) << right << j << endl;

192  fout.close(); cout << endl; coutCBL << "I wrote the file: " << file_out << endl;
