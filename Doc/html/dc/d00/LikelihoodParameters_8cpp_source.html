204 void cbl::statistics::LikelihoodParameters::set_parameters (const size_t nparameters, std::vector<ParameterType> parameterTypes, std::vector<std::string> parameterNames)
205 {
206  //Check parameterTypes size
207 
208  if (nparameters==0)
209  ErrorCBL("nparameters must be >0!", "set_parameters", "LikelihoodParameters.cpp");
210 
211  if ((parameterTypes.size()!=nparameters) && (parameterTypes.size()!=0))
212  ErrorCBL("the size of parameterTypes is incorrect!", "set_parameters", "LikelihoodParameters.cpp");
213 
214  if ((parameterNames.size()!=nparameters) && (parameterNames.size()!=0))
215  ErrorCBL("the size of parameterNames is incorrect!", "set_parameters", "LikelihoodParameters.cpp");
216 
217 
218  if ((parameterTypes.size()==nparameters) && (parameterNames.size()==nparameters)) {
219  m_nparameters=nparameters;
220  m_parameter_type = parameterTypes;
221  m_parameter_name = parameterNames;
222  }
223  else if ((parameterTypes.size()==0) && (parameterNames.size()==0)) {
224  m_nparameters=nparameters;
225  vector<ParameterType> pTypes(m_nparameters);
226  vector<string> pNames(m_nparameters);

228  pTypes[i] = ParameterType::_Base_;
229  pNames[i] = "par_"+conv(i+1, par::fINT);

231  m_parameter_type = pTypes;
232  m_parameter_name = pNames;
233  }
234  else if ((parameterTypes.size()==nparameters) && (parameterNames.size()==0)) {
235  m_nparameters=nparameters;
236  vector<string> pNames(m_nparameters);
237  for(size_t i=0; i<m_nparameters; i++)
238  pNames[i] = "par_"+conv(i+1, par::fINT);
239 
240  m_parameter_type = parameterTypes;
241  m_parameter_name = pNames;
242  }
243  else if ((parameterTypes.size()==0) && (parameterNames.size()==0)) {
244  m_nparameters=nparameters;
245  vector<ParameterType> pTypes(m_nparameters);
246  for(size_t i=0; i<m_nparameters; i++)
247  pTypes[i] = ParameterType::_Base_;
248 
249  m_parameter_type = pTypes;
250  m_parameter_name = parameterNames;
251  }
252 
253  m_parameter_bestfit_value.erase(m_parameter_bestfit_value.begin(), m_parameter_bestfit_value.end());
254  m_parameter_isFixed.resize(m_nparameters, false);
