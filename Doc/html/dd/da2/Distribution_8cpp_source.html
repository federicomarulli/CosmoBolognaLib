549 void cbl::glob::Distribution::get_distribution (vector<double> &xx, vector<double> &fx, vector<double> &err, const std::vector<double> FF, const std::vector<double> WW, const int nbin, const bool linear, const std::string file_out, const double fact, const double V1, const double V2, const bool bin_type, const bool conv, const double sigma)
550 {
551  if (xx.size()>0 || fx.size()>0 || FF.size()<=0 || nbin<=0) ErrorCBL("Error in cbl::glob::Distribution::get_distribution() in Distribution.cpp: the following conditions have to be satisfied: xx.size()<=0, fx.size()<=0, FF.size()>0 and nbin>0. The values recived are instead: xx.size() = " + cbl::conv(xx.size(), par::fINT) + ", fx.size() = " + cbl::conv(fx.size(), par::fINT) + ", FF.size() = " + cbl::conv(FF.size(), par::fINT) + "and nbin = " + cbl::conv(nbin, par::fINT) + "!");
552 
553  double minFF = (V1>cbl::par::defaultDouble) ? V1 : Min(FF)*0.9999;
554  double maxFF = (V2>cbl::par::defaultDouble) ? V2 : Max(FF)*1.0001;
555 
556 
557  // using GSL to create the histogram
558 
559  gsl_histogram *histo = gsl_histogram_alloc(nbin);
560 
561  if (linear) gsl_histogram_set_ranges_uniform(histo, minFF, maxFF);
562

564  vector<double> vv = logarithmic_bin_vector(nbin+1, minFF, maxFF);
565  double *vvv = new double[nbin+1]; for (int i=0; i<nbin+1; i++) vvv[i] = vv[i];
566  gsl_histogram_set_ranges(histo, vvv, nbin+1);
567  }
568 
569  vector<double> Weight = WW;
570  if (Weight.size()==0) Weight.resize(FF.size(), 1.);
571  checkDim(Weight, FF.size(), "WW");
572 
573  for (size_t i=0; i<FF.size(); i++)

575 
576  double x1, x2;
577 
578  for (int i=0; i<nbin; i++) {
579 
580  gsl_histogram_get_range(histo, i, &x1, &x2);
581  double val = gsl_histogram_get(histo, i);
582 
583  if (linear) xx.push_back(0.5*(x1+x2));
584  else xx.push_back(pow(10., 0.5*(log10(x1)+log10(x2))));
585 
586  if (bin_type) {
587  fx.push_back(val/((x2-x1)*fact));
588  err.push_back(sqrt(val)/((x2-x1)*fact));
589  }
590  else {
591  fx.push_back(val/((log10(x2)-log10(x1))*fact));
592  err.push_back(sqrt(val)/((log10(x2)-log10(x1))*fact));
593  }
594 
595  }
596 
597 
598  // Gaussian convolution
599 
600  if (conv) {
601  coutCBL << "The distribution is smoothed with a Gaussian filter" << endl;
602  double *func;
603  fftw_complex *func_tr;
604 
605  if (!linear) ErrorCBL("Work in progress...", ExitCode::_workInProgress_);
606  int nbinN = 2*nbin;
607  int i1 = nbin*0.5, i2 = 1.5*nbin;
608 
609  int nbinK = 0.5*nbinN+1;
610 
611  func = fftw_alloc_real(nbinN);
612  func_tr = fftw_alloc_complex(nbinK);
613 
614  for (int i=0; i<nbinN; i++)
615  func[i] = 0;
616 
617  for (int i=i1; i<i2; i++)
618  func[i] = fx[i-i1];
619 
620  for (int i=0; i<nbinK; i++) {
621  func_tr[i][0] = 0;
622  func_tr[i][1] = 0;
623  }
624

626  real2complex = fftw_plan_dft_r2c_1d(nbinN, func, func_tr, FFTW_ESTIMATE);
627  fftw_execute(real2complex);
628  fftw_destroy_plan(real2complex);
629 
630  double delta = (maxFF-minFF)/nbin;
631  double SS = pow(sigma,2);
632 
633  double fact = 2*par::pi/(nbinN*delta);
634  for (int i=0; i<nbinK; i++) {
635  double kk = i*fact;
636  func_tr[i][0] = func_tr[i][0]*exp(-0.5*kk*kk*SS);
637  func_tr[i][1] = func_tr[i][1]*exp(-0.5*kk*kk*SS);
638  }
639 
640  fftw_plan complex2real;
641  complex2real = fftw_plan_dft_c2r_1d(nbinN, func_tr, func, FFTW_ESTIMATE);
642  fftw_execute(complex2real);
643  fftw_destroy_plan(complex2real);
644 
645  for (int i=i1; i<i2; i++)
646  fx[i-i1] = func[i]/nbinN;
647  }
648 
649 
650  if (file_out!=par::defaultString && file_out!="") {
651 
652  ofstream fout(file_out.c_str()); checkIO(fout, file_out);
653 
654  for (size_t i=0; i<xx.size(); i++)
655  fout << xx[i] << " " << fx[i] << " " << err[i] << endl;
656

658  }
659 
660  gsl_histogram_free(histo);
