67 double cbl::measure::PowerSpectrum_angular::m_dtheta_theta (const BinType binType, int i)
68 { //Returns dtheta_i * theta_i
69  double value=0;
70  if (binType==cbl::BinType::_linear_) //dtheta * theta
71  value = (m_theta[1]-m_theta[0])*m_theta[i];
72 
73  else if (binType==cbl::BinType::_logarithmic_) // d log theta * theta^2 = d theta / theta * theta^2 = dtheta * theta
74  value = (log(m_theta[1])-log(m_theta[0]))*m_theta[i]*m_theta[i];
75 
76  return value;
77 }

137 std::vector<double> cbl::measure::PowerSpectrum_angular::m_error (const BinType binType, std::vector<double> w_err)
138 {
139  std::vector<double> Celle_err(m_ell.size(), 0.0);
140  double dlogell = log(m_ell[1])-log(m_ell[0]); //logarithmic bin amplitude
141  double p0 = 1/dlogell;
142  double d, pref, Gp, fp;
143  for (size_t j=0; j<Celle_err.size(); j++) {
144  for (size_t i=0; i<w_err.size();i++) {
145  d = m_dtheta_theta(binType, i);
146  pref = p0*d/(m_theta[i]*m_theta[i]);
147  Gp = m_theta[i]*m_ell[j]*sqrt(std::exp(dlogell))*cyl_bessel_j(1,m_theta[i]*m_ell[j]*sqrt(std::exp(dlogell)))-m_theta[i]*m_ell[j]/sqrt(std::exp(dlogell))*cyl_bessel_j(1,m_theta[i]*m_ell[j]/sqrt(std::exp(dlogell)));
148  fp = m_w_err[i]*Gp;
149  Celle_err[j] += pref*pref*fp*fp; //square for error propagation
150  }
151  }
152  for (size_t i=0; i<Celle_err.size(); i++) {
153  Celle_err[i] = sqrt(Celle_err[i]);
154  }
155  return Celle_err;
156 }
