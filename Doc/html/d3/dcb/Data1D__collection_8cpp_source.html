421  vector<double> vv;

423  if ( (m_x[i][j] < vxmin[i]) || (m_x[i][j]>vxmax[i]))

440  shared_ptr<Data> dd = make_shared<Data1D>(Data1D(xx[0], data, covariance));
441 
442  return dd;
