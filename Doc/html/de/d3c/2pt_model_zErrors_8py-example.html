This example shows how to compute the two-point correlation function model taking into account the effect of redshift errors

32

36

39 figure = plt.figure()

43 
44 ax.set_xscale("log")

46 
47 for ss in SigmaZ:
48  SigmaS = cbl.cc*ss*(1.+redshift)/cosmology.HH(redshift)
49  xi = cbl.damped_Xi(dv(rr), bias, linear_growth_rate, SigmaS, dv(kk), interpPk)

51  ax.plot(rr, xi, label=r"$\sigma_z = %g$"%ss)
