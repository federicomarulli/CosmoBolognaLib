180 void cbl::statistics::Posterior::maximize (const std::vector<double> start, const unsigned int max_iter, const double tol, const double epsilon)
181 {
182  vector<double> starting_par;
183 
184  unsigned int npar_free = m_model->parameters()->nparameters_free();
185  unsigned int npar = m_model->parameters()->nparameters();
186 
187  if (start.size()==npar_free)
188  starting_par = start;
189  else if (start.size()==npar)
190  for (size_t i=0; i<npar_free; i++)
191  starting_par.push_back(start[m_model->parameters()->free_parameter()[i]]);
192  else
193  ErrorCBL("check your inputs: start.size()="+conv(start.size(), par::fINT)+" must be equal to either npar_free="+conv(npar_free, par::fINT)+" or npar="+conv(npar, par::fINT)+"!", "maximize", "Posterior.cpp");
194 
195  function<double(vector<double> &)> post = [this](vector<double> &pp) { return -this->log(pp); };
196 
197 
198  // extra check on epsilon
199 
200  function<bool(vector<double> &)> checkWrong = [&] (vector<double> &pp)
201  {
202  bool ch = true;
203  if (post(pp)<-par::defaultDouble)
204  ch = false;
205  return ch;
206  };
207 
208  vector<double> par = starting_par;
209  if (checkWrong(par))
210  ErrorCBL("The starting position is outside the prior range: the first input parameter must be changed!", "maximize", "Posterior.cpp");
211 
212  // loop on simplex side
213  for (size_t i=0; i<npar_free; i++) {
214  par = starting_par;
215  par[i] += epsilon;
216  if (checkWrong(par))
217  ErrorCBL("The simplex side is outside prior range: the epsilon parameter or the starting position must be changed.", "maximize", "Posterior.cpp");
218  }
219 
220  // everything is fine up to here... let's go
221  coutCBL << "Maximizing the posterior..." << endl;
222  vector<double> result = cbl::wrapper::gsl::GSL_minimize_nD(post, starting_par, {}, max_iter, tol, epsilon);
223 
224  // check if the result is inside the prior ranges
225  if (m_prior->log(result)>par::defaultDouble) {
226  coutCBL << "Done!" << endl << endl;
227  m_model_parameters->set_bestfit_values(result);
228  m_model_parameters->write_bestfit_info();
229  coutCBL << "log(posterior) = " << post(result) << endl << endl;
230  }
231  else
232  ErrorCBL("the maximization ended with parameter values out of the priors: check your inputs or change the epsilon value!", "maximize", "Posterior.cpp");
233 
234 }

240 void cbl::statistics::Posterior::sample_stretch_move (const double aa, const bool parallel, const string outputFile, const int start, const int thin, const int nbins)
241 {
242  if (parallel && outputFile!=cbl::par::defaultString)
243  WarningMsgCBL("no run-time output available for parallel stretch-move algorithm: this option will be ignored!", "sample_stretch_move", "Posterior.cpp");
244 
245  coutCBL << "Sampling the posterior..." << endl;
246 
247  const int seed = m_generate_seed();
248  const int nparameters = m_model_parameters->nparameters();
249  const int nparameters_free = m_model_parameters->nparameters_free();
250  const int chain_size = m_model_parameters->chain_size();
251  const int n_walkers = m_model_parameters->chain_nwalkers();
252 
253  vector<vector<double>> Start(n_walkers, vector<double>(nparameters, 0));
254 
255  for (int i=0; i<nparameters; i++)
256  for (int j=0; j<n_walkers; j++)
257  Start[j][i] = m_model_parameters->chain_value(i, 0, j);
258 
259  auto posterior = [this] (vector<double> &pp) { return log(pp); };
260 
261  cbl::statistics::Sampler sampler(nparameters, nparameters_free, posterior);
262  if (parallel)
263  sampler.sample_stretch_move_parallel(chain_size, n_walkers, Start, seed, aa);
264  else
265  sampler.sample_stretch_move(chain_size, n_walkers, Start, seed, aa, outputFile);
266 
267  vector<vector<double>> chain_values;
268 
269  sampler.get_chain_function_acceptance(chain_values, m_log_posterior, m_acceptance);
270 
271  m_model_parameters->set_chain_values(chain_values, n_walkers);
272 
273  // set the best-fit parameters to the meadian values of the MCMC chain
274  m_model_parameters->set_bestfit_values(start, thin, nbins, m_generate_seed());
275 
276  m_weight.erase(m_weight.begin(), m_weight.end());
277  m_weight.resize(m_log_posterior.size(), 1.);
278 }

284 void cbl::statistics::Posterior::importance_sampling (const std::string input_dir, const std::string input_file, const vector<size_t> column, const int header_lines_to_skip, const bool is_FITS_format, const bool apply_to_likelihood, const int n_walkers)
285 {
286  // import the chain
287  read_chain(input_dir, input_file, n_walkers, column, header_lines_to_skip, is_FITS_format);
288 
289  const int n_parameters = m_model_parameters->nparameters();
290  const int chain_size = m_model_parameters->chain_size();
291 
292  vector<double> log_dist(m_log_likelihood.size(), 1.), input_log_dist(m_log_likelihood.size(), 1.);
293 
294  // loop on the chain
295  for (int chain_position=0; chain_position<chain_size; ++chain_position) {
296 
297  // loop to parallelize the computation
298 #pragma omp parallel for schedule(dynamic)
299  for (int walker_index=0; walker_index<n_walkers; ++walker_index) {
300 
301  vector<double> pp(n_parameters);
302  for (int parameter=0; parameter<n_parameters; ++parameter)
303  pp[parameter] = m_model_parameters->chain_value(parameter, chain_position, walker_index);
304 
305  // compute the logarithm of the posterior, of likelihood, at the parameters of the input chain
306  log_dist[chain_position*n_walkers+walker_index] = (apply_to_likelihood) ? Likelihood::log(pp) : log(pp);
307 
308  // value of the logarithm of the posterior, or likelihood, of the input chain
309  input_log_dist[chain_position*n_walkers+walker_index] =
310  (apply_to_likelihood) ? m_log_likelihood[chain_position*n_walkers+walker_index]
311  : m_log_posterior[chain_position*n_walkers+walker_index];
312  }
313 
314  coutCBL << int(double(chain_position)/(chain_size)*100.) << "% \r"; cout.flush();
315  }
316 
317 
318  // compute the weights as posterior, or likelihood, normalized ratios
319 
320  const double shift = Max(input_log_dist)-Max(log_dist);
321 
322  for (int chain_position=0; chain_position<chain_size; ++chain_position)
323 #pragma omp parallel for schedule(dynamic)
324  for (int walker_index=0; walker_index<n_walkers; ++walker_index)
325  m_weight[chain_position*n_walkers+walker_index] = exp(log_dist[chain_position*n_walkers+walker_index]-input_log_dist[chain_position*n_walkers+walker_index]+shift);
326 
327 }

345 void cbl::statistics::Posterior::read_chain_ascii (const string input_dir, const string input_file, const int n_walkers, const vector<size_t> column, const int skip_header)
346 {
347  const string file = input_dir+input_file;
348  coutCBL << "Reading the chain file " << file << endl;
349 
350  const int nparameters = m_model_parameters->nparameters();
351 
352  std::function<vector<double>(const vector<double> params)> assign;
353 
354  if (column.size()==0) {
355  assign = [&] (const vector<double> params) {
356  vector<double> pp(nparameters);
357  for (int i=0; i<nparameters; i++)
358  pp[i] = params[i+1];
359  return pp;
360  };
361  }
362  else {
363  checkDim(column, nparameters, "columns");
364  assign = [&] (const vector<double> params) {
365  vector<double> pp(nparameters);
366  for (int i=0; i<nparameters; i++)
367  pp[i] = params[column[i]+1];
368  return pp;
369  };
370  }
371 
372  ifstream fin(file.c_str()); checkIO(fin, file);
373 
374  string line;
375  for (int i=0; i<skip_header; i++)
376  getline(fin, line);
377 
378  vector<vector<double>> chain_value;
379 
380  m_log_likelihood.erase(m_log_likelihood.begin(), m_log_likelihood.end());
381  m_log_posterior.erase(m_log_posterior.begin(), m_log_posterior.end());
382  m_weight.erase(m_weight.begin(), m_weight.end());
383 
384  while (getline(fin, line)) {
385  stringstream ss(line);
386  double NUM;
387  vector<double> ll;
388 
389  while (ss>>NUM) ll.push_back(NUM);
390 
391  vector<double> params = assign(ll);
392 
393  m_log_likelihood.push_back(ll[ll.size()-4]);
394  m_log_posterior.push_back(ll[ll.size()-2]);
395  m_weight.push_back(ll[ll.size()-1]);
396 
397  chain_value.push_back(params);
398  }
399  const int chain_size = chain_value.size()/n_walkers;
400 
401  fin.clear(); fin.close();
402 
403  checkDim(chain_value, n_walkers*chain_size, m_model_parameters->nparameters(), "chain_value");
404  checkDim(m_log_posterior, n_walkers*chain_size, "m_log_posterior");
405  checkDim(m_weight, n_walkers*chain_size, "m_weight");
406 
407  chain_value = cbl::transpose(chain_value);
408 
409  m_model_parameters->set_chain_values(chain_value, n_walkers);
410 
411  coutCBL << "Done!" << endl << endl;
412 }

517

520  }
521  }

523  fout.clear(); fout.close();
524 
525  coutCBL << "I wrote the file: " << file << endl;

532 void cbl::statistics::Posterior::write_chain_fits (const string output_dir, const string output_file, const int start, const int thin)
533 {
534  const int nparameters = m_model_parameters->nparameters();
535  const int chain_size = m_model_parameters->chain_size();
536  const int n_walkers = m_model_parameters->chain_nwalkers();
537 
538  checkDim(m_log_posterior, n_walkers*chain_size, "m_log_posterior", false);
539 
540  string MK = "mkdir -p "+output_dir;
541  if (system(MK.c_str())) {}
542 
543  vector<string> names;
544  names.push_back("Step");
545  for (int k=0; k<nparameters; k++)
546  names.emplace_back(m_model_parameters->name(k));
547 
548  names.push_back("Log(Likelihood)");
549  names.push_back("Log(Prior)");
550  names.push_back("Log(Posterior)");
551  names.push_back("Weight");
552 
553  vector<vector<double>> value(nparameters+5);
554 
555  int n = 0;
556  for (int j=start; j<chain_size; j+=thin)
557  for (int i=0; i<n_walkers; i++) {
558  value[0].emplace_back(n);
559  vector<double> pp(nparameters);
560  for (int k=0; k<nparameters; k++) {
561  value[k+1].emplace_back(m_model_parameters->chain_value(k, j, i));
562  pp[k] = m_model_parameters->chain_value(k, j, i);
563  }
564 
565  if (m_log_likelihood.size()>size_t(j*n_walkers+i)) {
566  value[nparameters+1].emplace_back(m_log_likelihood[j*n_walkers+i]);
567  value[nparameters+2].emplace_back(m_log_posterior[j*n_walkers+i]-m_log_likelihood[j*n_walkers+i]);
568  }
569  else {
570  const double lpr = m_prior->log(pp);
571  value[nparameters+1].emplace_back(m_log_posterior[j*n_walkers+i]-lpr);
572  value[nparameters+2].emplace_back(lpr);
573  }
574 
575  value[nparameters+3].emplace_back(m_log_posterior[j*n_walkers+i]);
576  value[nparameters+4].emplace_back(m_weight[j*n_walkers+i]);
577  n ++;
578  }
579 
580  cbl::wrapper::ccfits::write_table_fits(output_dir, output_file, names, value);
581 
582 }

634 void cbl::statistics::Posterior::initialize_chains (const int chain_size, const int n_walkers, const std::string input_dir, const std::string input_file)
635 {
636  string last_step_file = input_dir+input_file+"_LastStep";
637  string get_last_step = "tail -n "+conv(n_walkers, par::fINT)+" "+input_dir+input_file+" > "+last_step_file;
638  if (system(get_last_step.c_str())) {}
639 
640  ifstream fin(last_step_file);
641  string line;
642 
643  vector<vector<double>> chain_value;
644 
645  while (getline(fin, line))
646  {
647  stringstream ss(line);
648  double NUM;
649  vector<double> ll, params;
650 
651  while (ss>>NUM) ll.push_back(NUM);
652  for (size_t i=1; i<ll.size()-3; i++)
653  params.push_back(ll[i]);
654 
655  chain_value.push_back(params);
656  }
657  fin.clear(); fin.close();
658 
659  string rm_last_step = "rm -r "+last_step_file;
660  if (system(rm_last_step.c_str())) {}
661 
662  checkDim(chain_value, n_walkers, m_model_parameters->nparameters(), "chain_from_LastStep_file");
663  chain_value = cbl::transpose(chain_value);
664 
665  initialize_chains(chain_size, chain_value);
666 }
