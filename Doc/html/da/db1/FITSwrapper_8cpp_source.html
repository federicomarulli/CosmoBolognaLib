48 vector<vector<double>> cbl::ccfitswrapper::read_table_fits (const string input_fits, const vector<string> column_names, const int next, const double fill_value)
49 {
50  ifstream check_input(input_fits); checkIO(check_input, input_fits); check_input.close();
51 
52  unique_ptr<CCfits::FITS> pInfile(new CCfits::FITS(input_fits, CCfits::Read, next));
53 
54  CCfits::ExtHDU& table = pInfile->currentExtension();
55 
56  const long nrows = table.rows();
57  size_t no_col = 0;
58 
59  if (nrows==0)
60  ErrorCBL("Error in read_table_fits, no rows in the selected table extension!");
61 
62  vector<vector<double>> cc;
63 
64  for (size_t i=0; i<column_names.size(); i++) {
65  vector<double> vv;
66  try {
67  table.column(column_names[i]).read(vv, 0, nrows);
68  }
69  catch (CCfits::Table::NoSuchColumn) {
70  if (fill_value==par::defaultDouble)
71  ErrorCBL("Error in read_table_fits! No column "+column_names[i]+"!");
72  else {
73  vv.erase(vv.begin(), vv.end());
74  vv.resize(nrows, fill_value);
75  no_col+=1;
76  }
77  }
78  cc.push_back(vv);
79  }
80 
81  if (no_col==column_names.size())
82  ErrorCBL("Error in read_table_fits, no column found!");
83 
84  return cc;
85 }
