68 shared_ptr<data::Data> cbl::measure::numbercounts::NumberCounts1D::m_measurePoisson ()
69 {
70  auto histogram = make_shared<glob::Histogram1D> (glob::Histogram1D ());
71  histogram->set(m_histogram->nbins(), m_histogram->minVar(), m_histogram->maxVar(), m_histogram->shift(), m_histogram->bin_type(), m_histogram->edges());
72 
73  histogram->put(m_data->var(m_Var), m_data->var(catalogue::Var::_Weight_));
74 
75  m_histogram = histogram;
76 
77  vector<double> bins = m_histogram->bins();
78  vector<double> edges = m_histogram->edges();
79  vector<double> hist = m_histogram->operator()(m_HistogramType, m_fact);
80  vector<double> error = m_histogram->error(m_HistogramType, m_fact);
81  vector<double> counts;
82  vector<double> counts_error;
83  vector<double> normalization;
84 
85  for (size_t i=0; i<m_histogram->nbins(); i++) {
86  counts.push_back(m_histogram->unweighted_counts(i));
87  counts_error.push_back(sqrt(static_cast<double>(counts[i])));
88  normalization.push_back(m_histogram->normalization(i, m_HistogramType, m_fact));
89  }
90 
91  vector<vector<double>> extra_info (7);
92  extra_info[0] = slice(edges, 0, m_histogram->nbins());
93  extra_info[1] = slice(edges, 1, m_histogram->nbins()+1);
94  extra_info[2] = counts;
95  extra_info[3] = counts_error;
96  extra_info[4] = normalization;
97  extra_info[5] = m_histogram->averaged_bins();
98  extra_info[6] = m_histogram->error_bins();
99 
100  auto dataset = make_shared<data::Data1D_extra> (data::Data1D_extra(bins, hist, error, extra_info));
101 
102  return dataset;
103 }

171  region_weights[i][ran()] ++;

178  for (size_t i=0; i<m_data->nObjects(); i++) {
179  int bin = m_histogram->digitize(var[i]);
180  for (int j=0; j<nResamplings; j++) {
181  histo_BS[j]->put (bin, weight[i]*region_weights[j][regions[i]], var[i]);
182  }
183  }

199  for (int i=0; i<nResamplings; i++)
200  histo_BS[i]->write(dir_output_resample, "Bootstrap_"+conv(i+1, par::fINT)+".dat", m_HistogramType, m_fact);

275 shared_ptr<data::Data> cbl::measure::numbercounts::NumberCounts1D::Gaussian_smoothing (const double sigma)
276 {
277  coutCBL << "The distribution is smoothed with a Gaussian filter" << endl;
278  double *func;
279  fftw_complex *func_tr;
280 
281  if (m_histogram->bin_type()!=BinType::_linear_) ErrorCBL("", "Gaussian_smoothing", "NumberCounts1D.cpp", glob::ExitCode::_workInProgress_);
282 
283  auto histogram = make_shared<glob::Histogram1D> (glob::Histogram1D ());
284  histogram->set(m_histogram->nbins(), m_histogram->minVar(), m_histogram->maxVar(), m_histogram->shift(), m_histogram->bin_type());
285 
286  int nbin = m_histogram->nbins();
287  int nbinN = 2*nbin;
288  int i1 = nbin*0.5, i2 = 1.5*nbin;
289 
290  int nbinK = 0.5*nbinN+1;
291 
292  func = fftw_alloc_real(nbinN);
293  func_tr = fftw_alloc_complex(nbinK);
294 
295  for (int i=0; i<nbinN; i++)
296  func[i] = 0;
297 
298  for (int i=i1; i<i2; i++)
299  func[i] = m_histogram->operator()(m_HistogramType, m_fact)[i-i1];
300 
301  for (int i=0; i<nbinK; i++) {
302  func_tr[i][0] = 0;
303  func_tr[i][1] = 0;
304  }
305 
306  fftw_plan real2complex;
307  real2complex = fftw_plan_dft_r2c_1d(nbinN, func, func_tr, FFTW_ESTIMATE);
308  fftw_execute(real2complex);
309  fftw_destroy_plan(real2complex);
310 
311  double delta = (m_histogram->maxVar()-m_histogram->minVar())/nbin;
312  double SS = pow(sigma,2);
313 
314  double fact = 2*par::pi/(nbinN*delta);
315  for (int i=0; i<nbinK; i++) {
316  double kk = i*fact;
317  func_tr[i][0] = func_tr[i][0]*exp(-0.5*kk*kk*SS);
318  func_tr[i][1] = func_tr[i][1]*exp(-0.5*kk*kk*SS);
319  }
320 
321  fftw_plan complex2real;
322  complex2real = fftw_plan_dft_c2r_1d(nbinN, func_tr, func, FFTW_ESTIMATE);
323  fftw_execute(complex2real);
324  fftw_destroy_plan(complex2real);
325 
326  vector<double> new_hist;
327 
328  for (int i=0; i<i2-i1; i++) new_hist.push_back(func[i+i1]/nbinN);
329 
330  auto dataset = make_shared<data::Data1D_extra> (data::Data1D_extra(m_histogram->bins(), new_hist, m_dataset->error(), m_dataset->extra_info()));
331 
332  //gsl_histogram_free(m_histogram);
333  fftw_cleanup();
334 
335  return dataset;
336 
337 }
