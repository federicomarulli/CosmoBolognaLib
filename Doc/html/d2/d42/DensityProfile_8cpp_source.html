118 double cbl::cosmology::Cosmology::density_profile (const double rad, const double Mass, const double redshift, const string model_cM, const string profile, const string halo_def) const
119 {
120  if (profile!="NFW")
121  return ErrorCBL("profile not allowed!", "density_profile", "DensityProfile.cpp");
122 
123  const double conc = concentration(Mass, redshift, model_cM, profile, halo_def);
124 
125  const double rho_s = rho_crit(redshift)*Delta_c(redshift)/3.*pow(conc, 3)/(log(1.+conc)-conc/(1.+conc));
126 
127  const double r_s = r_vir(Mass, redshift)/conc;
128 
129  return rho_s/((rad/r_s)*pow(1.+rad/r_s, 2));
130 }

170 double cbl::cosmology::Cosmology::Mass_Delta (const double Mass, const double Delta_in, const double Delta_out, const double conc, const bool is_input_conc, const double rRmin_guess, const double rRmax_guess) const
171 {
172  auto func = [&] (const double xx)
173  {
174  const double c_in = (is_input_conc) ? conc : conc*xx;
175  const double c_out = (!is_input_conc) ? conc : conc/xx;
176  const double AA = log(1.+c_out)-c_out/(1.+c_out);
177  const double BB = log(1.+c_in)-c_in/(1.+c_in);
178  return fabs(Delta_in/Delta_out*AA*pow(xx, 3)-BB);
179  };
180 
181  // R_[Delta_out]/R_[Delta_in]
182  const double Rratio = wrapper::gsl::GSL_minimize_1D(func, 1., max(1.e-3, rRmin_guess), rRmax_guess);
183 
184  // M_[Delta_out]
185  return Delta_out/Delta_in/pow(Rratio, 3)*Mass;
186 }
