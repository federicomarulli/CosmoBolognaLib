42 def go (data, model, nwalkers, chain_size, results_name, model_name):
43 
44  # define a Gaussian likelihood
45  likelihood = cbl.Likelihood(data, model, cbl.LikelihoodType__Gaussian_Error_)
46 
47  # set the model parameter
48  valA, valB, valC = 1., 2., 1.
49 
50  # limits for A and B
51  minA, maxA = -10., 10.
52  minB, maxB = -10., 10.
53 
54  # construct the priors
55  prior_A = cbl.PriorDistribution(cbl.DistributionType__Uniform_, minA, maxA)
56  prior_B = cbl.PriorDistribution(cbl.DistributionType__Uniform_, minB, maxB)
57  prior_C = cbl.PriorDistribution(cbl.DistributionType__Constant_, valC)
58  prior_distributions = cbl.PriorDistributionPtrVector([prior_A, prior_B, prior_C])
59 
60  # construct the posterior
61  posterior = cbl.Posterior(prior_distributions, likelihood, 696)
62 
63  # sample the posterior (starting the MCMC chain from the maximum of the posterior to speed up the chain convergence)
64  posterior.initialize_chains(chain_size, nwalkers, 1.e-5, [valA, valB])
65  posterior.sample_stretch_move(2)
66 
67  # show the median MCMC values of the four parameters on screen
68  print ("\n")
69  for i in range(posterior.parameters().nparameters()):
70  print("Posterior median of %s = %g\n"%(posterior.parameters().name(i), posterior.parameters().bestfit_value(i)))
71 
72  # store the chain ouputs
73  posterior.write_results("../output/", results_name)
74 
75  return posterior

79 def plot_contours (posterior, burn_in, thin, figure, axes, color, label):
80 
81  # read the chains
82  nPar = posterior.parameters().nparameters()
83  table = np.array( [posterior.parameters().parameter_chain_values(i, burn_in, thin) for i in range(nPar)] )
84  names = [posterior.parameters().name(i) for i in range(nPar)]
85  status = [posterior.parameters().status(i) for i in range(nPar)]
86  weights = posterior.weight(burn_in, thin)
87 
88  # select only non-fixed parameters
89  from itertools import compress
90  notFixed = list(map(lambda s : s!="FIXED", status))
91  notFixedNames = list(compress(names, notFixed))
92  notFixedChains = np.array(list(compress(table, notFixed))).T
93 
94  corner.corner(notFixedChains, \
95  weights=np.array(weights),\
96  labels=notFixedNames,\
97  truths=(1, 2., 4.), truth_color='black', \
98  plot_datapoints=False, show_titles=True, smooth=True, plot_density=False, levels=(1-np.exp(-0.5), 1-np.exp(-2), 1-np.exp(-4.5)),\
99  color=color, fill_contours=True, hist_kwargs={"density" : True, "color":color}, fig=figure)
100 
101  axes[0][1].plot(1,1,color, label=label)
102  axes[0][1].legend(loc="best", fontsize=11)
