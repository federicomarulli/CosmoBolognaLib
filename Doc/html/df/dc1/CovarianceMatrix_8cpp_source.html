63 void cbl::data::CovarianceMatrix::m_set (const vector<double> covariance, const double nmeasures, const double prec)
64 {
65  (void)prec;
66 
67  m_order = sqrt(covariance.size());
68 
69  m_hartlap_factor = hartlap_factor(m_order, nmeasures);
70 
71  m_matrix = cbl::wrapper::eigen::SquareMatrixToEigen(covariance);
72  m_precision = m_matrix.inverse();

74  m_variance = m_matrix.diagonal();

153 void cbl::data::CovarianceMatrix::read (const std::string filename, const int cov_col, const int skipped_lines, const double nmeasures, const double prec)
154 {
155  ifstream fin(filename.c_str()); checkIO(fin, filename);
156  string line;
157 
158  for (int i=0; i<skipped_lines; ++i) getline(fin, line);
159 
160  vector<double> covariance;
161 
162  while (getline(fin, line)) {
163 
164  stringstream ss(line);
165  vector<double> num; double NN = par::defaultDouble;
166  while (ss>>NN) num.push_back(NN);
167 
168  if (int(num.size())>=cov_col && num[cov_col]>par::defaultDouble)
169  covariance.push_back(num[cov_col]);
170  }
171 
172  fin.clear(); fin.close();
173 
174  m_set(covariance, nmeasures, prec);
175 }

203 void cbl::data::CovarianceMatrix::measure (const std::vector<std::shared_ptr<Data>> dataset, const double normalization, const double prec)
204 {
205  const size_t nbins = dataset[0]->ndata();
206  const size_t nmocks = dataset.size();
207 
208  vector<double> mean(nbins);
209 
210  for (size_t i=0; i<nbins; i++)
211  for (size_t j=0; j<nmocks; j++)
212  mean[i] = dataset[j]->data(i)/nmocks;
213 
214  double norm = normalization/(nmocks-1);
215 
216  vector<double> covariance;
217  for (size_t i=0; i<nbins; i++)
218  for (size_t j=0; j<nbins; j++) {
219  double val = 0;
220  for (size_t n=0; n<nmocks; n++)
221  val += (dataset[n]->data(i)-mean[i])*(dataset[n]->data(j)-mean[j])*norm;
222  covariance.push_back(val);
223  }
224 
225  m_set(covariance, nmocks, prec);
226 }

231 void cbl::data::CovarianceMatrix::measure(const std::vector<std::vector<std::shared_ptr<Data>>> dataset, const double normalization, const double prec)
232 {
233  size_t nmocks = dataset[0].size();
234  size_t nbins = 0;
235 
236  for (size_t i=0; i<dataset.size(); i++)
237  nbins += dataset[0][i]->ndata();
238 
239  m_hartlap_factor = 1-(nbins+1)/(nmocks-1);
240 
241  vector<vector<double>> table(nbins, vector<double>(nmocks, 0));
242  vector<double> mean(nbins);
243 
244  int start=0;
245  for (size_t i=0; i<dataset.size(); i++){
246  for (size_t j=0; j<dataset[i].size(); j++) {
247  for (int k=0; k<dataset[i][j]->ndata(); k++){
248  table[start+k][j] = dataset[i][j]->data(k);
249  mean[start+k] += dataset[i][j]->data(k)/nmocks;
250  }
251  }
252  start += dataset[i][0]->ndata();
253  }
254 
255  double norm = normalization/(nmocks-1);
256 
257  vector<double> covariance;
258  for (size_t i=0; i<nbins; i++)
259  for (size_t j=0; j<nbins; j++) {
260  double val = 0;
261  for (size_t n=0; n<nmocks; n++)
262  val += (table[i][n]-mean[i])*(table[j][n]-mean[j])*norm;
263  covariance.push_back(val);
264  }
265 
266  m_set(covariance, nmocks, prec);
267 }

274 {

277

294 cbl::data::CovarianceMatrix cbl::data::CovarianceMatrix::operator += (const cbl::data::CovarianceMatrix covariance) const
295 {
