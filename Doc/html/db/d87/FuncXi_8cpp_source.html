45 double cbl::glob::func_xi_GSL (double kk, void *params)
46 {
47  struct cbl::glob::STR_xi *pp = (struct cbl::glob::STR_xi *) params;
48 
49  double lgk = log10(kk);
50 
51  double lgPkK = interpolated(lgk, pp->lgkk, pp->lgPk, "Spline");
52

80 double cbl::xi_from_Pk (const double rr, const vector<double> lgkk, const vector<double> lgPk, const double k_min, const double k_max, const double aa, const double prec)
81 {

87  cbl::glob::STR_xi str;

95  Func.params = &str;

110 {
111  int C1 = c1-1, C2 = c2-1;

114

118

227 double cbl::sigmaR (const double RR, const int corrType, const vector<double> rr, const vector<double> corr)
228 {
229  double sigmaR = -1;
230 
231  if (corrType==1) { // using the spherically averaged correlation function
232 
233  auto ff = [&] (const double rad)
234  {
235  const double xiR = interpolated(rad, rr, corr, "Poly");
236  return (3.-2.25*rad/RR+0.1875*pow(rad/RR, 3))*rad*rad*xiR;
237  };
238 
239  const double prec = 0.0001;
240  const double Int = gsl::GSL_integrate_qaws (ff, 0., 2*RR, 1, 0, 0, 0, prec);
241 
242  if (1./pow(RR,3)*Int<0) ErrorCBL("Error in sigmaR with xi of Func.cpp!,"+conv(1./pow(RR,3)*Int,par::fDP4)+"<0");
243  sigmaR = sqrt(1./pow(RR,3)*Int);
244  }
245 
246  else if (corrType==2) { // using the projected correlation function
247 
248  auto ff = [&] (const double rad)
249  {
250  const double wpR = interpolated(rad, rr, corr, "Poly");
251 
252  const double xx = rad/RR;
253 
254  const double gg = (xx<=2)
255  ? 1./(2.*par::pi)*(3.*par::pi-9.*xx+pow(xx,3))
256  : 1./(2.*par::pi)*((-pow(xx,4)+11.*pow(xx,2)-28.)/sqrt(pow(xx,2)-4.)+pow(xx,3)-9.*xx+6.*asin(2./xx));
257 
258  return wpR*rad*gg;
259  };
260 
261  const double prec = 0.0001;
262  const double Int1 = gsl::GSL_integrate_qaws(ff, 0., 1., prec);
263  const double Int2 = gsl::GSL_integrate_qaws(ff, 1., 100., prec);
264 
265  if (1./pow(RR,3)*(Int1+Int2)<0) ErrorCBL("Error in sigmaR with wp of Func.cpp!,"+conv(1./pow(RR,3)*(Int1+Int2),par::fDP4)+"<0");
266  sigmaR = sqrt(1./pow(RR,3)*(Int1+Int2));
267  }
268 
269  else ErrorCBL("Error in sigmaR of Func.cpp!");
270 
271  return sigmaR;
272 }

331 double cbl::barred_xi_direct (const double RR, const vector<double> rr, const vector<double> xi, const double rAPP, const double r0, const double gamma)
332 {
333  vector<double> xi_;
334 
335  // power-law approximation for r<rAPP
336 
337  double kk = 0;
338  while (rr[kk]<rAPP)
339  xi_.push_back(3.*pow(rr[kk++]/r0,-gamma)/(-gamma+3.));
340  kk --;
341 
342 
343  // estimate the function at r>rApp
344 
345  double Int = (rAPP>0) ? xi_[kk]*pow(rr[kk],3)/3. : 0.;
346 
347  double bin = rr[1]-rr[0];
348 
349  for (unsigned int i=kk+1; i<rr.size(); i++) {
350  xi_.push_back(3./pow(rr[i],3)*(Int+xi[i]*pow(rr[i],2)*bin));
351  Int += xi[i]*pow(rr[i],2)*bin;
352  }
353 
354  return interpolated(RR, rr, xi_, "Linear");
355 }

362 {
363  vector<double> xi__;
364 
365 
366  // power-low approximation for r<rAPP
367 
368  double kk = 0;
369  while (rr[kk]<rAPP)
370  xi__.push_back(5.*pow(rr[kk++]/r0,-gamma)/(-gamma+5.));
371  kk --;

374  // estimate the function at r>rApp
375 
376  double Int = (rAPP>0) ? xi__[kk]*pow(rr[kk],5)/5. : 0.;
377 
378  double bin = rr[1]-rr[0];
379 
380  for (unsigned int i=kk+1; i<rr.size(); i++) {
381  xi__.push_back(5./pow(rr[i],5)*(Int+xi[i]*pow(rr[i],4)*bin));

392 double cbl::barred_xi_ (const double RR, const vector<double> rr, const vector<double> xi, const double rApp, const double r0, const double gamma)
393 {
394  vector<double> xi_;

445 
446

450  ErrorCBL(Err);

466  double xi_4 = multipole_xi4_model(beta, xi_real, xi_, xi__);

476 double cbl::xi2D_lin_model (const double beta, const double bias, const double xi_real, const double xi_, const double xi__, const double P_2, const double P_4)
477 {
478  double bias2 = bias*bias;

480  double Xi_ = xi_ * bias2;

494 double cbl::xi2D_lin_model (const double rp, const double pi, const double beta, const double bias, const vector<double> rad_real, const vector<double> xi_real, const vector<double> xi_, const vector<double> xi__, const int index, const bool bias_nl, const double bA)
495 {
496  double rr = sqrt(rp*rp+pi*pi);
497  double cos = pi/rr;
498 
499  double xiR = (index>-1) ? xi_real[index] : -1.;
500  double xiR_ = (index>-1) ? xi_[index] : -1.;
501  double xiR__ = (index>-1) ? xi__[index] : -1.;
502 
503  if (xiR<0) {
504  xiR = interpolated(rr, rad_real, xi_real, "Linear");
505  xiR_ = interpolated(rr, rad_real, xi_, "Linear");
506  xiR__ = interpolated(rr, rad_real, xi__, "Linear");

514  xiR_ *= bias2;

519  double xi_4 = multipole_xi4_model(beta, xiR, xiR_, xiR__);

529 double cbl::xi2D_model (double rp, double pi, shared_ptr<void> pp, vector<double> par)
530 {
531  (void)rp; (void)pi;
532 
533  if (par.size()<3) {
534  string Err = "Error in xi2D_model! par.size() = " + conv(par.size(),par::fINT) + "!";
535  ErrorCBL(Err);
536  }
537 
538  shared_ptr<cbl::glob::STR_xi2D_model> vec = static_pointer_cast<cbl::glob::STR_xi2D_model>(pp);
539 
540  int index = par[par.size()-1];
541  int ind_min = index*vec->step_v;
542  int ind_max = ind_min+vec->step_v;
543 
544  double sigma12 = par[1];
545 
546  vector<double> par2;
547  par2.push_back(par[0]);
548  par2.push_back(par[2]);
549  for (unsigned int i=3; i<par.size(); i++) par2.push_back(par[i]);
550 
551 
552  // convolution of the linear xi(rp,pi) with the velocity distribution function f(v)
553 
554  double xi2D_nl = 0;
555  double norm = 0.; // to avoid numerical problems when sigma12 is too small
556 
557  for (int i=ind_min; i<ind_max; i++) {
558  par2[par2.size()-1] = i;
559  xi2D_nl += xi2D_lin_model(vec->rp[i], vec->pi[i], pp, par2)*f_v(vec->vel[i], sigma12, vec->FV)*vec->delta_v;
560  norm += f_v(vec->vel[i], sigma12, vec->FV)*vec->delta_v;
561  }
562 
563  xi2D_nl /= norm;
564 
565 
566  if (fabs(norm-1)>0.1) {
567  string Warn = "Attention! sigma12 = "+conv(sigma12,par::fDP2)+" ---> norm = " + conv(norm,par::fDP3) + ", the number of bins used for the convolution with f(v) should be increased!";
568  WarningMsg(Warn);
569  print(par);
570  }
571 
572  return xi2D_nl;
573 }

580 double cbl::xi2D_model (const double rp, const double pi, const double beta, const double bias, const double sigma12, const vector<double> rad_real, const vector<double> xi_real, const vector<double> xi_, const vector<double> xi__, const double var, const int FV, int index, const bool bias_nl, const double bA, const double v_min, const double v_max, const int step_v)
581 {
582  double delta_v = (v_max-v_min)/step_v;
583 
584  double vel = v_min;
585 
586  double xi2D = 0.;
587 
588 
589  for (int k=0; k<step_v; k++) {
590 
591  double pi_new = pi-vel*var;
592 
593  double rr = sqrt(rp*rp+pi_new*pi_new);
594  double cos = pi_new/rr;
595 
596  double xiR = (index>-1) ? xi_real[index] : -1.;
597  double xiR_ = (index>-1) ? xi_[index] : -1.;
598  double xiR__ = (index>-1) ? xi__[index] : -1.;
599 
600  if (xiR<0) {
601  xiR = interpolated(rr, rad_real, xi_real, "Linear");
602  xiR_ = interpolated(rr, rad_real, xi_, "Linear");
603  xiR__ = interpolated(rr, rad_real, xi__, "Linear");
604  }
605 
606  double Bias = bias;
607  if (bias_nl) Bias *= b_nl(rr, bA);
608 
609  xi2D += xi2D_lin_model(beta, Bias, xiR, xiR_, xiR__, P_2(cos), P_4(cos))*f_v(vel, sigma12, FV)*delta_v;
610 
611  vel += delta_v;
612  if (index>-1) index ++;
613  }
614 
615  return xi2D;
616 }

633 double cbl::f_v (const double vel, const double rp, const double pi, const double var, const double sigmav0, const double cmu, const double cs1, const double cs2)
634 {
635 
636  double sp = sqrt(pow(rp,2)+pow(pi-vel*var,2));
637  double mup = 1./sp*(pi-vel*var);
638 
639  double sigmav = sigmav0*(1.+cmu*mup*mup)*(1.+cs1*exp(-cs2*rp*rp));
640 
641  return 1./(sigmav*sqrt(2.))*exp(-sqrt(2.)*fabs(vel)/sigmav);
