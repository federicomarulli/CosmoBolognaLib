122 void cbl::modelling::Modelling::set_likelihood (const statistics::LikelihoodType likelihood_type, const vector<size_t> x_index, const int w_index, const double prec, const int Nres)
123 {
124  if (m_model==NULL)
125  ErrorCBL("undefined model!", "set_likelihood", "Modelling.cpp");
126 
127  if (m_fit_range) {
128  if (m_data_fit==NULL)
129  ErrorCBL("undefined fit range!", "set_likelihood", "Modelling.cpp");
130  m_likelihood = make_shared<statistics::Likelihood> (statistics::Likelihood(m_data_fit, m_model, likelihood_type, x_index, w_index, NULL, prec, Nres));
131  }
132 
133  else {
134  if (m_data==NULL)
135  ErrorCBL("Error in set_likelihood of Modelling.cpp. Undefined dataset!", "set_likelihood", "Modelling.cpp");
136  m_likelihood = make_shared<statistics::Likelihood> (statistics::Likelihood(m_data, m_model, likelihood_type, x_index, w_index, NULL, prec, Nres));
137  }
