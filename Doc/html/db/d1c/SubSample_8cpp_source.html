430 void cbl::check_regions (catalogue::Catalogue &data, catalogue::Catalogue &random)
431 {
432  coutCBL << "Checking if the regions have been assigned correctly..." << endl;
433 
434  // check if data and random catalogues have the same number of regions
435  // nRegions is a "proxy" for the region geometry
436  if (data.nRegions()!=random.nRegions())
437  ErrorCBL("data and random have different number of regions: data_regions.size() = "+conv(data.nRegions(), par::fINT)+", random_regions.size = "+conv(random.nRegions(), par::fINT)+": please set the number of regions through set_region_number() or check your inputs!", "check_regions", "GlobalFunc/SubSample.cpp");
438 
439  size_t nRegions = data.nRegions();
440 
441  // count how many objects fall in the regions
442  vector<long> dataObj_region(nRegions, 0);
443  vector<long> data_region = data.region();
444 
445  for (size_t i=0; i<data.nObjects(); ++i) {
446  checkDim(dataObj_region, data_region[i], "dataObj_region", false);
447  dataObj_region[data_region[i]] ++;
448  }
449 
450  vector<long> randObj_region(nRegions, 0);
451  vector<long> rand_region = random.region();
452  for (size_t i=0; i<random.nObjects(); ++i) {
453  checkDim(randObj_region, rand_region[i], "randObj_region", false);
454  randObj_region[rand_region[i]] ++;
455  }
456 
457  // empty random regions are not allowed!
458  // rearrange regions
459  size_t region_eff = 0;
460  std::map<long, long> region_list_eff;
461  for (size_t i=0; i<nRegions; i++) {
462  if (randObj_region[i]!=0) {
463  region_list_eff.insert(pair<long, long>(i, region_eff));
464  region_eff ++;
465  }
466  else {
467  // throw error if random region is empty and data region is not
468  if (dataObj_region[i]!=0)
469  ErrorCBL("the "+conv(i, par::fINT)+" region is empty in the random sample, but contains "+conv(dataObj_region[i], par::fINT)+" data points; this is not allowed, Please check your inputs!", "check_regions", "GlobalFunc/SubSample.cpp");
470  }
471  }
472 
473  if (region_eff!=nRegions) {
474  coutCBL << "Found "+conv(region_eff, par::fINT)+" non-empty regions" << endl;
475  coutCBL << "Rearranging regions in data and random sample..." << endl;
476  for (size_t i=0; i<data.nObjects(); ++i) {
477  checkDim(data_region, i, "data_region", false);
478  if (int(region_list_eff.size())<data_region[i]) ErrorCBL("the dimension of region_list_eff is: " + conv(region_list_eff.size(), par::fINT) + " ( < " + conv(data_region[i], par::fINT) + " )", "check_regions", "SubSample.cpp");
479  data_region[i] = region_list_eff[data_region[i]];
480  }
481 
482  for (size_t i=0; i<random.nObjects(); ++i) {
483  checkDim(rand_region, i, "rand_region", false);
484  if (int(region_list_eff.size())<rand_region[i]) ErrorCBL("the dimension of region_list_eff is: " + conv(region_list_eff.size(), par::fINT) + " ( < " + conv(rand_region[i], par::fINT) + " )", "check_regions", "SubSample.cpp");
485  rand_region[i] = region_list_eff[rand_region[i]];
486  }
487 
488  data.set_region(data_region, region_eff+1);
489  random.set_region(rand_region, region_eff+1);
490  coutCBL << "Done!" << endl;
491  }
492 
493  coutCBL << "End check regions!" << endl;
494 
495 }
