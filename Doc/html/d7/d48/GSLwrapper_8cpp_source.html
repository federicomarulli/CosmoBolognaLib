431 double cbl::wrapper::gsl::GSL_root_brent (gsl_function Func, const double low_guess, const double up_guess, const double rel_err, const double abs_err)
432 {
433  gsl_set_error_handler_off();
434 
435  int status = 0, iter = 0, max_iter = 10000;
436  const gsl_root_fsolver_type *T;
437  double r;
438 
439  gsl_root_fsolver *s;
440  T = gsl_root_fsolver_brent;
441  s = gsl_root_fsolver_alloc(T);
442 
443  double x_lo = low_guess;
444  double x_hi = up_guess;
445 
446  gsl_root_fsolver_set(s, &Func, x_lo, x_hi);
447 
448  do
449  {
450  iter ++;
451  status = gsl_root_fsolver_iterate(s);
452 
453  if ((status!=GSL_SUCCESS) && (status!=GSL_CONTINUE))
454  check_GSL_fail(status, true, "GSL_root_brent", "gsl_root_fsolver_iterate");
455 
456  r = gsl_root_fsolver_root(s);
457  x_lo = gsl_root_fsolver_x_lower(s);
458  x_hi = gsl_root_fsolver_x_upper(s);
459 
460  status = gsl_root_test_interval(x_lo, x_hi, abs_err, rel_err);
461 
462  if ((status!=GSL_SUCCESS) && (status!=GSL_CONTINUE))
463  check_GSL_fail(status, true, "GSL_root_brent", "gsl_root_test_interval");
464  }
465 
466  while (status==GSL_CONTINUE && iter<max_iter);
467 
468  gsl_root_fsolver_free(s);
469 
470  check_GSL_fail(status, true, "GSL_minimize_nD", par::defaultString);
471 
472  return r;
473 }

542

570 }
