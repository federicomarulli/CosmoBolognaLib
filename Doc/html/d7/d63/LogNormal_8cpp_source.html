84 std::vector<double> cbl::lognormal::LogNormal::get_xi_model(const std::vector<double> radius)
85 {
86  vector<double> stat;
87  m_data->stats_var(Var::_Redshift_, stat);
88  vector<double> PkG;
89  vector<double> kG = linear_bin_vector(500, -4., 2.);
90 
91  double ff = m_cosmology->linear_growth_rate(stat[0], 0.);
92  double beta = ff/m_bias;
93  double fact = pow(m_bias,2);
94  fact = (m_Real) ? fact : fact*(1+2.*beta/3.+0.2*beta*beta);
95 
96  for (size_t i=0; i<kG.size(); i++) {
97  kG[i] = pow(10,kG[i]);
98  PkG.push_back(fact*m_cosmology->Pk_DM(kG[i], m_author, m_NL, stat[0], true, m_model));
99  }
100 
101  return cbl::wrapper::fftlog::transform_FFTlog (radius, 1, kG, PkG, 0, 0, 2.*par::pi, 1);
102 }
