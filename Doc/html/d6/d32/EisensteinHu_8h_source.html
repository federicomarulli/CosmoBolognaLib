233  int TFmdm_set_cosm (double omega_matter, double omega_baryon, double omega_hdm, int degen_hdm, double omega_lambda, double hubble, double redshift, double As=2.56e-9, double k_pivot=0.05, double n_spec=0.96)
234  {
235  double z_drag_b1, z_drag_b2, omega_denom;
236  int qwarn;
237  qwarn = 0;
238 
239  theta_cmb = 2.728/2.7; // assuming T_cmb = 2.728 K
240 
241  /* Look for strange input */
242 
243  if (omega_baryon<0.0) {
244  fprintf(stderr,
245  "TFmdm_set_cosm(): Negative omega_baryon set to trace amount.\n");
246  qwarn = 1;
247  }
248 
249  if (omega_hdm<0.0) {
250  fprintf(stderr,
251  "TFmdm_set_cosm(): Negative omega_hdm set to trace amount.\n");
252  qwarn = 1;
253  }
254 
255  if (hubble<=0.0) {
256  fprintf(stderr,"TFmdm_set_cosm(): Negative Hubble constant illegal.\n");
257  exit(1); /* Can't recover */
258  } else if (hubble>2.0) {
259  fprintf(stderr,"TFmdm_set_cosm(): Hubble constant should be in units of 100 km/s/Mpc.\n");
260  qwarn = 1;
261  }
262 
263  if (redshift<=-1.0) {
264  fprintf(stderr,"TFmdm_set_cosm(): Redshift < -1 is illegal.\n");
265  exit(1);
266  } else if (redshift>99.0) {
267  fprintf(stderr,
268  "TFmdm_set_cosm(): Large redshift entered. TF may be inaccurate.\n");
269  qwarn = 1;
270  }
271 
272  if (degen_hdm<1) degen_hdm = 1;
273  num_degen_hdm = (float) degen_hdm;
274  /* Have to save this for TFmdm_onek_mpc() */
275  /* This routine would crash if baryons or neutrinos were zero,
276  so don't allow that */
277  if (omega_baryon<=0) omega_baryon=1e-5;
278  if (omega_hdm<=0) omega_hdm=1e-5;
279 
280  omega_curv = 1.0-omega_matter-omega_lambda;
281  omhh = omega_matter*pow(hubble, 2);
282  obhh = omega_baryon*pow(hubble, 2);
283  onhh = omega_hdm*pow(hubble, 2);
284  f_baryon = omega_baryon/omega_matter;
285  f_hdm = omega_hdm/omega_matter;
286  f_cdm = 1.0-f_baryon-f_hdm;
287  f_cb = f_cdm+f_baryon;
288  f_bnu = f_baryon+f_hdm;
289 
290  /* Compute the equality scale. */
291  z_equality = 25000.0*omhh*pow(theta_cmb, -4); /* Actually 1+z_eq */
292  k_equality = 0.0746*omhh*pow(theta_cmb, -2);
293 
294  /* Compute the drag epoch and sound horizon */
295  z_drag_b1 = 0.313*pow(omhh,-0.419)*(1+0.607*pow(omhh,0.674));
296  z_drag_b2 = 0.238*pow(omhh,0.223);
297  z_drag = 1291*pow(omhh,0.251)/(1.0+0.659*pow(omhh,0.828))*(1.0+z_drag_b1*pow(obhh,z_drag_b2));
298  y_drag = z_equality/(1.0+z_drag);
299 
300  sound_horizon_fit = 44.5*log(9.83/omhh)/sqrt(1.0+10.0*pow(obhh,0.75));
301 
302  /* Set up for the free-streaming & infall growth function */
303  p_c = 0.25*(5.0-sqrt(1+24.0*f_cdm));
304  p_cb = 0.25*(5.0-sqrt(1+24.0*f_cb));
305 
306  omega_denom = omega_lambda+pow(1.0+redshift, 2)*(omega_curv+omega_matter*(1.0+redshift));
307  omega_lambda_z = omega_lambda/omega_denom;
308  omega_matter_z = omega_matter*pow(1.0+redshift,2)*(1.0+redshift)/omega_denom;
309 
310  growth_k0 = z_equality/(1.0+redshift)*2.5*omega_matter_z/(pow(omega_matter_z,4.0/7.0)-omega_lambda_z+(1.0+omega_matter_z/2.0)*(1.0+omega_lambda_z/70.0));
311  growth_to_z0 = z_equality*2.5*omega_matter/(pow(omega_matter,4.0/7.0)-omega_lambda + (1.0+omega_matter/2.0)*(1.0+omega_lambda/70.0));
312  double growth_z = 1./(1.0+redshift)*2.5*omega_matter_z/(pow(omega_matter_z,4.0/7.0)-omega_lambda_z + (1.0+omega_matter_z/2.0)*(1.0+omega_lambda_z/70.0));
313  growth_to_z0 = growth_k0/growth_to_z0;
314 
315  /*
316  growth_k0 = z_equality*5.*omega_matter_z/(2*(1+redshift))/(1./70.+209./140.*omega_matter_z-pow(omega_matter_z,2)/140.+pow(omega_matter_z,4./7.));
317  growth_to_z0 = z_equality*5.*omega_matter_z/(2*(1+redshift))/(1./70.+209./140.*omega_matter_z-pow(omega_matter_z,2)/140.+pow(omega_matter_z,4./7.));
318  growth_to_z0 = growth_k0/growth_to_z0;
319  */
320 
321  /* Compute small-scale suppression */
322  alpha_nu = f_cdm/f_cb*(5.0-2.*(p_c+p_cb))/(5.-4.*p_cb)*
323  pow(1+y_drag,p_cb-p_c)*
324  (1+f_bnu*(-0.553+0.126*f_bnu*f_bnu))/
325  (1-0.193*sqrt(f_hdm*num_degen_hdm)+0.169*f_hdm*pow(num_degen_hdm,0.2))*
326  (1+(p_c-p_cb)/2*(1+1/(3.-4.*p_c)/(7.-4.*p_cb))/(1+y_drag));
327  alpha_gamma = sqrt(alpha_nu);
328  beta_c = 1/(1-0.949*f_bnu);
329 
330  /* Done setting scalar variables */
331  hhubble = hubble; /* Need to pass Hubble constant to TFmdm_onek_hmpc() */
332 
333  ns = n_spec;
334 
335  //double norm = 2.*par::pi*par::pi*m_scalar_amp*pow(2.*pow(par::cc/100.,2)/(5*m_Omega_matter)*DD(redshift),2)*pow(k/m_scalar_pivot, m_n_spec-1)*k_ov_h*pow(fact, -3);
336  pk_normalization = 2.*par::pi*par::pi*As*pow(2.*pow(par::cc/100.,2)/(5*omega_matter)*growth_z,2)*pow(1./k_pivot, ns-1);
337 
338  return qwarn;
339  }

353  double TFmdm_onek_mpc (double kk)
354  {
355  double tf_sup_L, tf_sup_C;
356  double temp1, temp2;
357 
358  qq = kk/omhh*pow(theta_cmb, 2);
359 
360  /* Compute the scale-dependent growth functions */
361  y_freestream = 17.2*f_hdm*(1+0.488*pow(f_hdm,-7.0/6.0))*pow(num_degen_hdm*qq/f_hdm, 2);
362  temp1 = pow(growth_k0, 1.0-p_cb);
363  temp2 = pow(growth_k0/(1+y_freestream),0.7);
364  growth_cb = pow(1.0+temp2, p_cb/0.7)*temp1;
365  growth_cbnu = pow(pow(f_cb,0.7/p_cb)+temp2, p_cb/0.7)*temp1;
366 
367  /* Compute the master function */
368  gamma_eff =omhh*(alpha_gamma+(1-alpha_gamma)/(1+pow(kk*sound_horizon_fit*0.43, 4)));
369  qq_eff = qq*omhh/gamma_eff;
370 
371  tf_sup_L = log(2.71828+1.84*beta_c*alpha_gamma*qq_eff);
372  tf_sup_C = 14.4+325/(1+60.5*pow(qq_eff,1.11));
373  tf_sup = tf_sup_L/(tf_sup_L+tf_sup_C*pow(qq_eff, 2));
374 
375  qq_nu = 3.92*qq*sqrt(num_degen_hdm/f_hdm);
376  max_fs_correction = 1+1.2*pow(f_hdm,0.64)*pow(num_degen_hdm,0.3+0.6*f_hdm)/(pow(qq_nu,-1.6)+pow(qq_nu,0.8));
377  tf_master = tf_sup*max_fs_correction;
378 
379  /* Now compute the CDM+HDM+baryon transfer functions */
380  tf_cb = tf_master*growth_cb/growth_k0;
381  tf_cbnu = tf_master*growth_cbnu/growth_k0;
382 
383  return tf_cb;
384  }
