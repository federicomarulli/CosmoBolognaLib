56 void cbl::measure::threept::ThreePointCorrelation_comoving_multipoles_single::m_count_triplets (std::vector<double> &NNN, std::vector<double> &RRR, const double r12_min, const double r12_max, const double r13_min, const double r13_max, const int norders, const catalogue::Catalogue& catalogue) const
57 {
58  // timer
59  time_t start; time(&start);
60  int dp = cout.precision();
61  cout.setf(ios::fixed); cout.setf(ios::showpoint); cout.precision(2);
62 
63  const int nObjects = catalogue.nObjects();
64 
66  double NN, RR;
67 
68  NNN.erase(NNN.begin(), NNN.end());
69  NNN.resize(norders, 0);
70 
71  RRR.erase(RRR.begin(), RRR.end());
72  RRR.resize(norders, 0);
73 
74  auto cc = make_shared<Catalogue>(catalogue);
75  chainmesh::ChainMesh_Catalogue cm;
76 
77  cm.set_par(r13_max*0.5, cc, r13_max);
78 
79  auto cat = cm.catalogue();
80 
81  // thread number
82  int tid = 0;
83 
84  // start the multithreading parallelization
85 #pragma omp parallel num_threads(omp_get_max_threads()) private(tid)
86  {
87  tid = omp_get_thread_num();
88 
89  SphericalHarmonics_Coefficients sph(norders, 2);
90  SphericalHarmonics_Coefficients sph_random(norders, 2);
91  vector<double> _zeta(norders, 0);
92  vector<double> _zeta_random(norders, 0);
93  double _nn=0;
94  double _rr=0;
95 
96  // parallelized loop
97 #pragma omp for schedule(static, 2)
98  for (int i=0; i<nObjects; i++)
99  {
100  sph.reset();
101 
102  double ixx = cat->xx(i);
103  double iyy = cat->yy(i);
104  double izz = cat->zz(i);
105  double iww = cat->weight(i);
106 
107  if(iww<0) // iww<0 RANDOM OBJECT!
108  sph_random.reset();
109 
110  vector<long> close_objects = cm.close_objects({ixx, iyy, izz}, -1);
111 
112  for (size_t j=0; j<close_objects.size(); j++) {
113 
114  int inCommon = 0;
115  int j_idx = close_objects[j];
116 
117  double jxx = cat->xx(j_idx);
118  double jyy = cat->yy(j_idx);
119  double jzz = cat->zz(j_idx);
120  double jww = cat->weight(j_idx);
121 
122  double xx = jxx-ixx;
123  double yy = jyy-iyy;
124  double zz = jzz-izz;
125 
126  double rr = sqrt(xx*xx+yy*yy+zz*zz);
127 
128  if (rr>r12_min && rr<r12_max){
129  vector<complex<double>> _alm = sph.alm(xx/rr, yy/rr, zz/rr);
130  sph.add (_alm, jww, 0);
131  if(iww<0 && jww <0)
132  sph_random.add(_alm, jww, 0);
133  inCommon += 1;
134  }
135 
136  if (rr>r13_min && rr<r13_max){
137  vector<complex<double>> _alm = sph.alm(xx/rr, yy/rr, zz/rr);
138  sph.add (_alm, jww, 1);
139  if(iww<0 && jww <0)
140  sph_random.add(_alm, jww, 1);
141  inCommon += 1;
142  }
143 
145  if (inCommon>1) {
146  _nn += iww*jww*jww;
147  if(iww<0 && jww <0)
148  _rr -= iww*jww*jww;
149  }
150  }
151 
152  for (int l=0; l<norders; l++){
153  _zeta[l] += iww*sph.power(l, 0, 1);
154  if(iww<0)
155  _zeta_random[l] -= iww*sph_random.power(l, 0, 1);
156  }
157 
158  // estimate the computational time and update the time count
159  if (i==int(nObjects*0.25)) coutCBL << ".............25% completed" << endl;
160  if (i==int(nObjects*0.5)) coutCBL << ".............50% completed" << endl;
161  if (i==int(nObjects*0.75)) coutCBL << ".............75% completed"<< endl;
162  }
163 
164 #pragma omp critical
165  {
166  NN += _nn;
167  RR += _rr;
168  for (int l=0; l<norders; l++){
169  NNN[l] += _zeta[l];
170  RRR[l] += _zeta_random[l];
171  }
172  }
173 
174  }
175 
176  for (int l=0; l<norders; l++){
177  NNN[l] -= NN;
178  RRR[l] -= RR;
179  }
180 
181  // show the time spent by the method
182  time_t end; time (&end);
183  double diff = difftime(end, start);
184  if (tid==0) {
185  if (diff<60) coutCBL << " time spent to count the triplets: " << diff << " seconds" << endl ;
186  else if (diff<3600) coutCBL << " time spent to count the triplets: " << diff/60 << " minutes" << endl ;
187  else coutCBL << " time spent to count the triplets: " << diff/3600 << " hours" << endl ;
188  }
189  cout.unsetf(ios::fixed); cout.unsetf(ios::showpoint); cout.precision(dp);
190 
191 }

289 void cbl::measure::threept::ThreePointCorrelation_comoving_multipoles_single::measure (const ErrorType errorType, const std::string dir_output_triplets, const std::vector<std::string> dir_input_triplets, const int nResamplings, const bool count_triplets, const bool tcount, const int seed)
290 {
291  coutCBL << "I'm computing the three-point correlation multipoles..." << endl;
292 
293  (void)errorType;
294  (void)nResamplings;
295  (void)tcount;
296  (void)seed;
297 
298  if (count_triplets) { // count triplets
299 
300  m_nnn.resize(m_nOrders, 0);
301  m_rrr.resize(m_nOrders, 0);
302 
303  size_t start, end;
304  for (size_t i=0; i<m_nSplit; i++) {
305 
306  if (m_nSplit>1)
307  coutCBL << "Computing the three-point correlation multipoles, " << i+1 << "-th splitting" << endl;
308 
309  start = (m_splitFactor<0) ? 0 : m_data->nObjects()*(i*m_splitFactor);
310  end = (m_splitFactor<0) ? m_random->nObjects() : m_data->nObjects()*((i+1)*m_splitFactor);
311 
312  auto joined = m_join_catalogues(*m_data, *m_random, start, end);
313 
314  vector<double> nnn, rrr;
315  m_count_triplets(nnn, rrr, m_r12Min, m_r12Max, m_r13Min, m_r13Max, m_nOrders, *joined);
316 
317  for (size_t j=0; j<nnn.size(); j++) {
318  m_nnn[j] += nnn[j];
319  m_rrr[j] += rrr[j];
320  }
321  }
322 
323  for (size_t i=0; i<m_nnn.size(); i++) {
324  m_nnn[i] /= m_nSplit;
325  m_rrr[i] /= m_nSplit;
326  }
327 
328  m_write_triplets(m_nnn, dir_output_triplets, "nnn_legendre.dat");
329  m_write_triplets(m_rrr, dir_output_triplets, "rrr_legendre.dat");
330  }
331  else { //Read triplets from file
332  m_read_triplets(m_nnn, dir_input_triplets, "nnn_legendre.dat");
333  m_read_triplets(m_rrr, dir_input_triplets, "rrr_legendre.dat");
334  }
335 
337 
338  m_zeta = m_SzapudiSzalay_multipoles (m_nnn, m_rrr);
339 
340  coutCBL << "Done!" << endl;
341 }

369 void cbl::measure::threept::ThreePointCorrelation_comoving_multipoles_single::resum (const std::string dir, const std::string file, const cbl::triplets::TripletType tripletType, const int nBins, const bool bin) const
370 {
371  std::function<double(const double, const double, const int)> legFunction;
372 
373  double min, max;
374  string name;
375 
376  switch (tripletType) {
377 
378  case triplets::TripletType::_comoving_theta_:
379  {
380  min = 0;
381  max = par::pi;
382  name = "theta";
383 
384  if (bin)
385  legFunction = [&] (const double min, const double max, const int ell) {return cbl::Legendre_polynomial_theta_average(min, max, ell); };
386  else
387  legFunction = [&] (const double min, const double max, const int ell) {return cbl::legendre_polynomial(cos(0.5*(min+max)), ell); };
388 
389  break;
390  }
391 
392  case triplets::TripletType::_comoving_costheta_:
393  {
394  min = -1;
395  max = 1.;
396  name = "costheta";
397 
398  if (bin)
399  legFunction = [&] (const double min, const double max, const int ell) {return cbl::Legendre_polynomial_mu_average(min, max, ell); };
400  else
401  legFunction = [&] (const double min, const double max, const int ell) {return cbl::legendre_polynomial(0.5*(min+max), ell); };
402 
403  break;
404  }
405 
406  case triplets::TripletType::_comoving_side_:
407  {
408  if (!bin) {
409  WarningMsgCBL("Cannot use bin=false: Legendre polynomial averaging will be forced!", "resum", "ThreePointCorrelation_comoving_multipoles_single.cpp");
410  }
411 
412  name = "r23";
413  min = (m_r12Min != m_r13Min) ? m_r13Min-m_r12Max : 0.;
414  max = m_r13Max+m_r12Max;
415  legFunction = [&] (const double min, const double max, const int ell) { return
416  Legendre_polynomial_triangles_average (m_r12Min, m_r12Max, m_r13Min, m_r13Max, min, max, ell, 1.e-3, 0, 1000); };
417  break;
418  }
419 
420  default:
421  {
422  ErrorCBL("wrong tripletType in input!", "resum", "ThreePointCorrelation_comoving_multipoles_single.cpp");
423  break;
424  }
425  }
426 
427  double delta = (max-min)/nBins;
428 
429  string outFile = dir+file;
430 
431  ofstream fout(outFile.c_str());
432 
433  fout << "# r12 r13 " << name << " zeta1 zeta2 nnn rrr" << endl;
434 
435  for (int i=0; i<nBins; i++) {
436  double _min = min+i*delta;
437  double _max = min+(i+1)*delta;
438 
439  double nnn=0., rrr=0., zeta=0.;
440 
441  for (size_t ell=0; ell<m_nOrders; ell++) {
442  double legVal = legFunction(_min, _max, ell);
443  double norm = legFunction(_min, _max, 0);
444 
445  nnn += legVal*m_nnn[ell]*0.5*(2.*ell+1)/norm;
446  rrr += legVal*m_rrr[ell]*0.5*(2.*ell+1)/norm;
447  zeta += legVal*m_zeta[ell]/norm;
448  }
449 
450  fout << 0.5*(m_r12Max+m_r12Min) << " " << 0.5*(m_r13Max+m_r13Min) << " " << (_max+_min)*0.5 << " " << zeta << " " << nnn/rrr << " " << nnn << " " << rrr << endl;
451  }
452 
453  fout.clear(); fout.close();
454 
455  coutCBL << "I wrote the file " << outFile << endl;
456 }
