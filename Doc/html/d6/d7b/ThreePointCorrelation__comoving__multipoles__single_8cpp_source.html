370 void cbl::measure::threept::ThreePointCorrelation_comoving_multipoles_single::resum (const std::string dir, const std::string file, const cbl::triplets::TripletType tripletType, const int nBins, const bool bin) const
371 {
372  std::function<double(const double, const double, const int)> legFunction;
373 
374  double min, max;
375  string name;
376 
377  switch (tripletType) {
378 
379  case triplets::TripletType::_comoving_theta_:
380  {
381  min = 0;
382  max = par::pi;
383  name = "theta";
384 
385  if (bin)
386  legFunction = [&] (const double min, const double max, const int ell) {return cbl::Legendre_polynomial_theta_average(min, max, ell); };
387  else
388  legFunction = [&] (const double min, const double max, const int ell) {return cbl::legendre_polynomial(cos(0.5*(min+max)), ell); };
389 
390  break;
391  }
392 
393  case triplets::TripletType::_comoving_costheta_:
394  {
395  min = -1;
396  max = 1.;
397  name = "costheta";
398 
399  if (bin)
400  legFunction = [&] (const double min, const double max, const int ell) {return cbl::Legendre_polynomial_mu_average(min, max, ell); };
401  else
402  legFunction = [&] (const double min, const double max, const int ell) {return cbl::legendre_polynomial(0.5*(min+max), ell); };
403 
404  break;
405  }
406 
407  case triplets::TripletType::_comoving_side_:
408  {
409  if (!bin) {

411  }
412 
413  name = "r23";
414  min = (m_r12Min != m_r13Min) ? m_r13Min-m_r12Max : 0.;
415  max = m_r13Max+m_r12Max;
416  legFunction = [&] (const double min, const double max, const int ell) { return
417  Legendre_polynomial_triangles_average (m_r12Min, m_r12Max, m_r13Min, m_r13Max, min, max, ell, 1.e-3, 0, 1000); };
418  break;
419  }
420 
421  default:
422  {
423  ErrorCBL("wrong tripletType in input!", "resum", "ThreePointCorrelation_comoving_multipoles_single.cpp");
424  break;
425  }
426  }
427 
428  double delta = (max-min)/nBins;
429 
430  string outFile = dir+file;
431 
432  ofstream fout(outFile.c_str());
433 
434  fout << "# r12 r13 " << name << " zeta1 zeta2 nnn rrr" << endl;
435 
436  for (int i=0; i<nBins; i++) {
437  double _min = min+i*delta;
438  double _max = min+(i+1)*delta;
439 
440  double nnn=0., rrr=0., zeta=0.;
441 
442  for (size_t ell=0; ell<m_nOrders; ell++) {
443  double legVal = legFunction(_min, _max, ell);
444  double norm = legFunction(_min, _max, 0);
445

447  rrr += legVal*m_rrr[ell]*0.5*(2.*ell+1)/norm;
